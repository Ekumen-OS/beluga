80  : params_{std::move(params)}, landmark_map_{std::move(landmark_map)} {}

88  [[nodiscard]] auto operator()(measurement_type&& detections) const {
89  return [this, detections = std::move(detections)](const state_type& state) -> weight_type {
90  Sophus::SE3d robot_pose_in_world;
91 
92  if constexpr (std::is_same_v<state_type, Sophus::SE3d>) {
93  // The robot pose state is already given in 3D,
94  robot_pose_in_world = state;
95  } else {
96  // The robot pose state is given in 2D. Notice that in this case
97  // the 2D pose of the robot is assumed to be that of the robot footprint (projection of the robot
98  // on the z=0 plane of the 3D world frame). This is so that we can tie the sensor reference frame
99  // to the world frame where the landmarks are given without additional structural information.
100  robot_pose_in_world = Sophus::SE3d{
101  Sophus::SO3d::rotZ(state.so2().log()),
102  Eigen::Vector3d{state.translation().x(), state.translation().y(), 0.0}};
103  }
104 
105  const auto detection_weight = [this, &robot_pose_in_world](const auto& detection) {
106  // calculate range and detection_bearing_in_robot to the detection from the robot
107  // the detection is already in robot frame
108  const auto& detection_category = detection.category;
109  const auto& detection_position_in_robot = detection.detection_position_in_robot;
110 
111  const auto detection_range_in_robot = detection_position_in_robot.norm();
112  const auto detection_bearing_in_robot = detection_position_in_robot.normalized();
113 
114  // convert the detection to the world frame to query the map
115  const auto detection_position_in_world = robot_pose_in_world * detection_position_in_robot;
116 
117  // find the closest matching landmark in the world map
118  const auto opt_landmark_position_in_world =
119  landmark_map_.find_nearest_landmark(detection_position_in_world, detection_category);
120 
121  // if we did not find a matching landmark, return 0.0
122  if (!opt_landmark_position_in_world) {
123  return 0.0;
124  }
125 
126  // convert landmark pose to world frame
127  // ignore height, because we are modelling the detection in 2D
128  const auto& landmark_position_in_world = *opt_landmark_position_in_world;
129  const auto landmark_in_robot_position = robot_pose_in_world.inverse() * landmark_position_in_world;
130 
131  const auto landmark_range_in_robot = landmark_in_robot_position.norm();
132  const auto landmark_bearing_in_robot = landmark_in_robot_position.normalized();
133 
134  // calculate the aperture angle between the detection and the landmark
135  const auto cos_aperture = landmark_bearing_in_robot.dot(detection_bearing_in_robot);
136  const auto sin_aperture = landmark_bearing_in_robot.cross(detection_bearing_in_robot).norm();
137  const auto bearing_error = std::atan2(sin_aperture, cos_aperture);
138 
139  const auto range_error = detection_range_in_robot - landmark_range_in_robot;
140 
141  // apply the error model from Probabilistic Robotics
142  const auto range_error_prob =
143  std::exp(-range_error * range_error / (2. * params_.sigma_range * params_.sigma_range));
144  const auto bearing_error_prob =
145  std::exp(-bearing_error * bearing_error / (2. * params_.sigma_bearing * params_.sigma_bearing));
146 
147  // We'll assume the probability of identification error to be zero
148  return range_error_prob * bearing_error_prob;
149  };
150 
151  return std::transform_reduce(detections.cbegin(), detections.cend(), 1.0, std::multiplies{}, detection_weight);
152  };
153  }
154 
156  void update_map(map_type&& map) { landmark_map_ = std::move(map); }
157 
158  private:
159  param_type params_;
160  LandmarkMap landmark_map_;
161 };
