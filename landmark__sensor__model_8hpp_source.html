118  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
119  Sophus::SE3d robot_pose_in_world;
120 
121  if constexpr (std::is_same_v<state_type, Sophus::SE3d>) {
122  // The robot pose state is already given in 3D,
123  robot_pose_in_world = state;
124  } else {
125  // The robot pose state is given in 2D. Notice that in this case
126  // the 2D pose of the robot is assumed to be that of the robot footprint (projection of the robot
127  // on the z=0 plane of the 3D world frame). This is so that we can tie the sensor reference frame
128  // to the world frame where the landmarks are given without additional structural information.
129  robot_pose_in_world = Sophus::SE3d{
130  Sophus::SO3d::rotZ(state.so2().log()),
131  Eigen::Vector3d{state.translation().x(), state.translation().y(), 0.0}};
132  }
133 
134  const auto detection_weight = [this, &robot_pose_in_world](const auto& detection) {
135  // calculate range and detection_bearing_in_robot to the detection from the robot
136  // the detection is already in robot frame
137  const auto& detection_category = detection.category;
138  const auto& detection_position_in_robot = detection.detection_position_in_robot;
139 
140  const auto detection_range_in_robot = detection_position_in_robot.norm();
141  const auto detection_bearing_in_robot = detection_position_in_robot.normalized();
142 
143  // convert the detection to the world frame to query the map
144  const auto detection_position_in_world = robot_pose_in_world * detection_position_in_robot;
145 
146  // find the closest matching landmark in the world map
147  const auto opt_landmark_position_in_world =
148  landmark_map_.find_nearest_landmark(detection_position_in_world, detection_category);
149 
150  // if we did not find a matching landmark, return 0.0
151  if (!opt_landmark_position_in_world) {
152  return 0.0;
153  }
154 
155  // convert landmark pose to world frame
156  // ignore height, because we are modelling the detection in 2D
157  const auto& landmark_position_in_world = *opt_landmark_position_in_world;
158  const auto landmark_in_robot_position = robot_pose_in_world.inverse() * landmark_position_in_world;
159 
160  const auto landmark_range_in_robot = landmark_in_robot_position.norm();
161  const auto landmark_bearing_in_robot = landmark_in_robot_position.normalized();
162 
163  // calculate the aperture angle between the detection and the landmark
164  const auto cos_aperture = landmark_bearing_in_robot.dot(detection_bearing_in_robot);
165  const auto sin_aperture = landmark_bearing_in_robot.cross(detection_bearing_in_robot).norm();
166  const auto bearing_error = std::atan2(sin_aperture, cos_aperture);
167 
168  const auto range_error = detection_range_in_robot - landmark_range_in_robot;
169 
170  // apply the error model from Probabilistic Robotics
171  const auto range_error_prob =
172  std::exp(-range_error * range_error / (2. * params_.sigma_range * params_.sigma_range));
173  const auto bearing_error_prob =
174  std::exp(-bearing_error * bearing_error / (2. * params_.sigma_bearing * params_.sigma_bearing));
175 
176  // We'll assume the probability of identification error to be zero
177  const auto prob = range_error_prob * bearing_error_prob;
178 
179  // TODO(unknown): We continue to use the sum-of-cubes formula that nav2 uses
180  // See https://github.com/Ekumen-OS/beluga/issues/153
181  return prob * prob * prob;
182  };
183 
184  return std::transform_reduce(points_.cbegin(), points_.cend(), 1.0, std::plus{}, detection_weight);
185  }
