83  [[nodiscard]] auto operator()(measurement_type&& detections) const {
84  return [this, detections = std::move(detections)](const state_type& state) -> weight_type {
85  Sophus::SE3d robot_pose_in_world;
86 
87  if constexpr (std::is_same_v<state_type, Sophus::SE3d>) {
88  // The robot pose state is already given in 3D,
89  robot_pose_in_world = state;
90  } else {
91  // The robot pose state is given in 2D. Notice that in this case
92  // the 2D pose of the robot is assumed to be that of the robot footprint (projection of the robot
93  // on the z=0 plane of the 3D world frame). This is so that we can tie the sensor reference frame
94  // to the world frame where the landmarks are given without additional structural information.
95  robot_pose_in_world = Sophus::SE3d{
96  Sophus::SO3d::rotZ(state.so2().log()),
97  Eigen::Vector3d{state.translation().x(), state.translation().y(), 0.0}};
98  }
99 
100  // precalculate the sensor pose in the world frame
101  const auto sensor_pose_in_world = robot_pose_in_world * params_.sensor_pose_in_robot;
102 
103  const auto detection_weight = [this, &sensor_pose_in_world](const auto& detection) {
104  // find the landmark the most closely matches the sample bearing vector
105  const auto opt_landmark_bearing_in_sensor = landmark_map_.find_closest_bearing_landmark(
106  detection.detection_bearing_in_sensor, detection.category, sensor_pose_in_world);
107 
108  // if we did not find a matching landmark, return 0.0
109  if (!opt_landmark_bearing_in_sensor) {
110  return 0.0;
111  }
112 
113  // recover sample bearing vector
114  const auto detection_bearing_in_sensor = detection.detection_bearing_in_sensor.normalized();
115 
116  // recover landmark bearing vector
117  const auto& landmark_bearing_in_sensor = *opt_landmark_bearing_in_sensor;
118 
119  // calculate the aperture angle between the detection and the landmark
120  const auto cos_aperture = detection_bearing_in_sensor.dot(landmark_bearing_in_sensor);
121  const auto sin_aperture = detection_bearing_in_sensor.cross(landmark_bearing_in_sensor).norm();
122 
123  // calculate the angle between the sample and the landmark
124  const auto bearing_error = std::atan2(sin_aperture, cos_aperture);
125 
126  // model the probability of the landmark being detected as depending on the bearing error
127  const auto bearing_error_prob =
128  std::exp(-bearing_error * bearing_error / (2. * params_.sigma_bearing * params_.sigma_bearing));
129 
130  return bearing_error_prob;
131  };
132 
133  return std::transform_reduce(detections.cbegin(), detections.cend(), 1.0, std::multiplies{}, detection_weight);
134  };
135  }
136 
138  void update_map(map_type&& map) { landmark_map_ = std::move(map); }
139 
140  private:
141  param_type params_;
142  LandmarkMap landmark_map_;
143 };
