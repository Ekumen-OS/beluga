117  [[nodiscard]] auto operator()(measurement_type&& points) const {
118  return [this, points = std::move(points)](const state_type& state) -> weight_type {
119  const auto transform = world_to_likelihood_field_transform_ * state;
120  const auto x_offset = transform.translation().x();
121  const auto y_offset = transform.translation().y();
122  const auto cos_theta = transform.so2().unit_complex().x();
123  const auto sin_theta = transform.so2().unit_complex().y();
124  const auto unknown_space_occupancy_prob = static_cast<float>(1. / params_.max_laser_distance);
125  return std::transform_reduce(
126  points.cbegin(), points.cend(), 1.0, std::plus{},
127  [this, x_offset, y_offset, cos_theta, sin_theta, unknown_space_occupancy_prob](const auto& point) {
128  // Transform the end point of the laser to the grid local coordinate system.
129  // Not using Eigen/Sophus because they make the routine x10 slower.
130  // See `benchmark_likelihood_field_model.cpp` for reference.
131  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
132  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
133  const auto pz =
134  static_cast<double>(likelihood_field_.data_near(x, y).value_or(unknown_space_occupancy_prob));
135  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
136  // See https://github.com/Ekumen-OS/beluga/issues/153
137  return pz * pz * pz;
138  });
139  };

158  static ValueGrid2<float> make_likelihood_field(const LikelihoodFieldModelParam& params, const OccupancyGrid& grid) {
159  const auto squared_distance = [&grid](std::size_t first, std::size_t second) {
160  return static_cast<float>((grid.coordinates_at(first) - grid.coordinates_at(second)).squaredNorm());
161  };
162 
163  const auto squared_max_distance = static_cast<float>(params.max_obstacle_distance * params.max_obstacle_distance);
164  const auto truncate_to_max_distance = [squared_max_distance](auto squared_distance) {
165  return std::min(squared_distance, squared_max_distance);
166  };
167 
169  const double two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit;
170  assert(two_squared_sigma > 0.0);
171 
172  const double amplitude = params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi()));
173  assert(amplitude > 0.0);
174 
175  const double offset = params.z_random / params.max_laser_distance;
176 
177  const auto to_likelihood = [amplitude, two_squared_sigma, offset](double squared_distance) {
178  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
179  };
180 
181  const auto neighborhood = [&grid](std::size_t index) { return grid.neighborhood4(index); };
182 
183  // determine distances to obstacles and calculate likelihood values in-place
184  // to minimize memory usage when dealing with large maps
185  auto distance_map = nearest_obstacle_distance_map(grid.obstacle_mask(), squared_distance, neighborhood);
186 
187  if (params.model_unknown_space) {
188  const double inverse_max_distance = 1 / params.max_laser_distance;
189  const double background_distance = -two_squared_sigma * std::log((inverse_max_distance - offset) / amplitude);
190 
191  distance_map |= beluga::actions::overlay(grid.unknown_mask(), background_distance);
192  }
193 
194  auto likelihood_values = std::move(distance_map) | //
195  ranges::actions::transform(truncate_to_max_distance) | //
196  ranges::actions::transform(to_likelihood);
197 
198  return ValueGrid2<float>{std::move(likelihood_values), grid.width(), grid.resolution()};
199  }
