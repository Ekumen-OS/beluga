54 template <typename MeanDerived, typename CovarianceDerived, typename TransferFn>
55 auto unscented_transform(
56  const Eigen::MatrixBase<MeanDerived>& mean,
57  const Eigen::MatrixBase<CovarianceDerived>& covariance,
58  TransferFn&& transfer_fn,
59  std::optional<typename MeanDerived::Scalar> kappa = std::nullopt) {
60  using Scalar = typename MeanDerived::Scalar;
61  static_assert(
62  std::is_same_v<typename MeanDerived::Scalar, typename CovarianceDerived::Scalar>,
63  "Mean and covariance scalar types differ.");
64  static_assert(std::is_floating_point_v<Scalar>);
65  static_assert(MeanDerived::ColsAtCompileTime == 1, "Mean should be a column vector.");
66  static_assert(
67  CovarianceDerived::ColsAtCompileTime == CovarianceDerived::RowsAtCompileTime, "Covariance matrix is not square.");
68  constexpr int kNin = MeanDerived::RowsAtCompileTime;
69  static_assert(kNin >= 1, "Input dimension should be greater than zero.");
70  static_assert(CovarianceDerived::ColsAtCompileTime == kNin);
71  const double k = kappa.value_or(std::max(kNin - 3, 0));
72  assert(k >= 0);
73  const Scalar w0 = static_cast<Scalar>(k) / static_cast<Scalar>(kNin + k);
74  std::vector<Eigen::Vector<Scalar, kNin>> sigma_points;
75  std::vector<Scalar> weights;
76  sigma_points.reserve(2 * kNin + 1);
77  weights.resize(2 * kNin + 1);
78  sigma_points.emplace_back(mean);
79  weights[0] = w0;
80  const Scalar wn = 1. / (2. * (kNin + k));
81 
82  // Compute the Cholesky decomposition of the covariance matrix
83  const auto llt = covariance.llt();
84  assert(llt.info() == Eigen::ComputationInfo::Success);
85  const Eigen::Matrix<Scalar, kNin, kNin> l_matrix = llt.matrixL();
86  // Precompute sqrt{n + kaappa} * L for speed sake.
87  const Eigen::Matrix<Scalar, kNin, kNin> scaled_l_matrix = std::sqrt(kNin + k) * l_matrix;
88 
89  for (int i = 0; i < kNin; ++i) {
90  sigma_points.emplace_back(mean + scaled_l_matrix.col(i));
91  sigma_points.emplace_back(mean - scaled_l_matrix.col(i));
92  }
93  std::fill(weights.begin() + 1, weights.end(), wn);
94 
95  const auto transformed_sigma_points = sigma_points | ranges::views::transform(transfer_fn) | ranges::to<std::vector>;
96 
97  constexpr int kNout = decltype(transfer_fn(mean))::RowsAtCompileTime;
98  static_assert(decltype(transfer_fn(mean))::ColsAtCompileTime == 1, "Output mean should be a column vector");
99 
100  const Eigen::Vector<Scalar, kNout> out_mean = std::transform_reduce(
101  transformed_sigma_points.begin(), transformed_sigma_points.end(), weights.begin(),
102  Eigen::Vector<Scalar, kNout>::Zero().eval(), std::plus<>{}, std::multiplies<>{});
103 
104  const Eigen::Matrix<Scalar, kNout, kNout> out_cov = std::transform_reduce(
105  transformed_sigma_points.begin(), transformed_sigma_points.end(), weights.begin(),
106  Eigen::Matrix<Scalar, kNout, kNout>::Zero().eval(), std::plus<>{},
107  [&](const auto& sigma_point, const auto weight) {
108  const Eigen::Vector<Scalar, kNout> error = sigma_point - out_mean;
109  return (weight * (error * error.transpose())).eval();
110  });
111 
112  return std::make_pair(out_mean, out_cov);
113 }
