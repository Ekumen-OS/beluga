53  template <class ExecutionPolicy, class Range, class LikelihoodRange>
54  constexpr void apply_impl(ExecutionPolicy&& policy, Range& range, const LikelihoodRange& likelihoods) const {
55  // Update particle weights using std::transform.
56  auto weights = range | beluga::views::weights;
57  std::transform(
58  policy, std::begin(weights), std::end(weights), std::begin(likelihoods), std::begin(weights),
59  [](auto weight, auto likelihood) { return weight * likelihood; });
60 
61  // Store raw likelihood if the particle type has storage for it.
62  // We use ranges::views::zip here because we are performing side-effects (writing to members)
63  // which std::transform is not designed for.
64  using ParticleType = ranges::range_value_t<Range>;
65  if constexpr (has_likelihood_member_v<ParticleType>) {

67  std::for_each(policy, std::begin(zipped), std::end(zipped), [](auto&& tuple) {
68  auto& particle = std::get<0>(tuple);
