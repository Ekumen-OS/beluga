45 struct CellHasher {
47  size_t operator()(const Eigen::Matrix<int, N, 1> vector) const {
48  size_t seed = 0;
49  for (auto i = 0L; i < vector.size(); ++i) {
50  auto elem = *(vector.data() + i);
51  seed ^= std::hash<int>()(elem) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
52  }
53  return seed;
54  }
55 };

58 template <int NDim, typename Scalar = double>
59 inline NDTCell<NDim, Scalar> fit_points(const std::vector<Eigen::Matrix<Scalar, NDim, 1>>& points) {
60  static constexpr double kMinVariance = 1e-5;
61  Eigen::Map<const Eigen::Matrix<Scalar, NDim, Eigen::Dynamic>> points_view(
62  reinterpret_cast<const Scalar*>(points.data()), 2, static_cast<int64_t>(points.size()));
63  const Eigen::Matrix<Scalar, NDim, 1> mean = points_view.rowwise().mean();
64  const Eigen::Matrix<Scalar, NDim, Eigen::Dynamic> centered = points_view.colwise() - mean;
65  // Use sample covariance.
66  Eigen::Matrix<Scalar, NDim, Eigen::Dynamic> cov =
67  (centered * centered.transpose()) / static_cast<double>(points_view.cols() - 1);
68  cov(0, 0) = std::max(cov(0, 0), kMinVariance);
69  cov(1, 1) = std::max(cov(1, 1), kMinVariance);
70  if constexpr (NDim == 3) {
71  cov(2, 2) = std::max(cov(2, 2), kMinVariance);
72  }
73  return NDTCell<NDim, Scalar>{mean, cov};
74 }

79 template <int NDim, typename Scalar = double>
80 inline std::vector<NDTCell<NDim, Scalar>> to_cells(
81  const std::vector<Eigen::Matrix<Scalar, NDim, 1>>& points,
82  const double resolution) {
83  static constexpr int kMinPointsPerCell = 5;
84 
85  Eigen::Map<const Eigen::Matrix<Scalar, NDim, Eigen::Dynamic>> points_view(
86  reinterpret_cast<const Scalar*>(points.data()), NDim, static_cast<int64_t>(points.size()));
87 
88  std::vector<NDTCell<NDim, Scalar>> ret;
89  ret.reserve(static_cast<size_t>(points_view.cols()) / kMinPointsPerCell);
90 
91  std::unordered_map<Eigen::Matrix<int, NDim, 1>, std::vector<Eigen::Matrix<Scalar, NDim, 1>>, CellHasher<NDim>>
92  cell_grid;
93  for (const Eigen::Matrix<Scalar, NDim, 1>& col : points) {
94  cell_grid[(col / resolution).template cast<int>()].emplace_back(col);
95  }
96 
97  for (const auto& [cell, points_in_cell] : cell_grid) {
98  if (points_in_cell.size() < kMinPointsPerCell) {
99  continue;
100  }
101  ret.push_back(fit_points<NDim, Scalar>(points_in_cell));
102  }
103 
104  return ret;
105 }

161 template <typename SparseGridT>
162 class NDTSensorModel {
163  public:
164  static_assert(std::is_same_v<SparseValueGrid<typename SparseGridT::map_type>, SparseGridT>);
166  using ndt_cell_type = typename SparseGridT::mapped_type;
167  static_assert(ndt_cell_type::num_dim == 2, "NDT sensor model is only implemented for 2D problems.");
169  using state_type = std::conditional_t<
170  ndt_cell_type::num_dim == 2,
171  Sophus::SE2<typename ndt_cell_type::scalar_type>,
172  Sophus::SE3<typename ndt_cell_type::scalar_type>>;
174  using weight_type = double;
176  using measurement_type = std::vector<Eigen::Matrix<typename ndt_cell_type::scalar_type, ndt_cell_type::num_dim, 1>>;
178  using map_type = SparseGridT;
180  using param_type = NDTModelParam<ndt_cell_type::num_dim>;
181 
183 
189  NDTSensorModel(param_type params, SparseGridT cells_data)
190  : params_{std::move(params)}, cells_data_{std::move(cells_data)} {
191  assert(params_.minimum_likelihood >= 0);
192  }
193 
195 
200  [[nodiscard]] auto operator()(measurement_type&& points) const {
201  return [this, cells = detail::to_cells<ndt_cell_type::num_dim, typename ndt_cell_type::scalar_type>(
202  points, cells_data_.resolution())](const state_type& state) -> weight_type {
203  return std::transform_reduce(
204  cells.cbegin(), cells.cend(), 1.0, std::plus{},
205  [this, state](const ndt_cell_type& ndt_cell) { return likelihood_at(state * ndt_cell); });
206  };
207  }
208 
211  [[nodiscard]] double likelihood_at(const ndt_cell_type& measurement) const {
212  double likelihood = 0;
213  const typename map_type::key_type measurement_cell = cells_data_.cell_near(measurement.mean);
214  for (const auto& offset : params_.neighbors_kernel) {
215  const auto maybe_ndt = cells_data_.data_at(measurement_cell + offset);
216  if (maybe_ndt.has_value()) {
217  likelihood += maybe_ndt->likelihood_at(measurement, params_.d1, params_.d2);
218  }
219  }
220  return std::max(likelihood, params_.minimum_likelihood);
221  }
222 
223  private:
224  const param_type params_;
225  const SparseGridT cells_data_;
226 };

240 template <typename NDTMapRepresentationT>
241 NDTMapRepresentationT load_from_hdf5_2d(const std::filesystem::path& path_to_hdf5_file) {
242  static_assert(std::is_same_v<typename NDTMapRepresentationT::mapped_type, NDTCell2d>);
243  static_assert(std::is_same_v<typename NDTMapRepresentationT::key_type, Eigen::Vector2i>);
244  if (!std::filesystem::exists(path_to_hdf5_file) || std::filesystem::is_directory(path_to_hdf5_file)) {
245  std::stringstream ss;
246  ss << "Couldn't find a valid HDF5 file at " << path_to_hdf5_file;
247  throw std::invalid_argument(ss.str());
248  }
249 
250  H5::H5File file(path_to_hdf5_file, H5F_ACC_RDONLY);
251  H5::DataSet means_dataset = file.openDataSet("means");
252  H5::DataSet covariances_dataset = file.openDataSet("covariances");
253  H5::DataSet resolution_dataset = file.openDataSet("resolution");
254  H5::DataSet cells_dataset = file.openDataSet("cells");
255 
256  std::array<hsize_t, 2> dims_out;
257  means_dataset.getSpace().getSimpleExtentDims(dims_out.data(), nullptr);
258 
259  Eigen::Matrix2Xd means_matrix(dims_out[1], dims_out[0]);
260  Eigen::Matrix2Xi cells_matrix(dims_out[1], dims_out[0]);
261 
262  means_dataset.read(means_matrix.data(), H5::PredType::NATIVE_DOUBLE);
263  cells_dataset.read(cells_matrix.data(), H5::PredType::NATIVE_INT);
264 
265  std::vector<Eigen::Array<double, 2, 2>> covariances(dims_out[0]);
266  covariances_dataset.read(covariances.data(), H5::PredType::NATIVE_DOUBLE);
267 
268  double resolution;
269 
270  resolution_dataset.read(&resolution, H5::PredType::NATIVE_DOUBLE);
271 
272  typename NDTMapRepresentationT::map_type map{};
273 
274  // Note: Ranges::zip_view doesn't seem to work in old Eigen.
275  for (size_t i = 0; i < covariances.size(); ++i) {
276  map[cells_matrix.col(static_cast<Eigen::Index>(i))] =
277  NDTCell2d{means_matrix.col(static_cast<Eigen::Index>(i)), covariances.at(i)};
278  }
279 
280  return NDTMapRepresentationT{std::move(map), resolution};
281 }
