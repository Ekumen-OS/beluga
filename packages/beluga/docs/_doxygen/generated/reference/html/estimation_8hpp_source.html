59 template <
60  class QuaternionRange,
61  class WeightsRange,
62  class Scalar = typename ranges::range_value_t<QuaternionRange>::Scalar>
63 Eigen::Quaternion<Scalar> weighted_average_quaternion(
64  const QuaternionRange& quaternions,
65  const WeightsRange& normalized_weights) {
66  // Implementation is based on https://github.com/strasdat/Sophus/blob/main/sophus/average.hpp#L133 with the variant of
67  // non-uniform weights.
68  // See https://ntrs.nasa.gov/api/citations/20070017872/downloads/20070017872.pdf equation (13).
69  const size_t num_quaternions = std::size(quaternions);
70  assert(num_quaternions >= 1);
71  Eigen::Matrix<Scalar, 4, Eigen::Dynamic> all_quaternions(4, num_quaternions);
72 
73  for (const auto& [i, quaternion, normalized_weight] :
74  ranges::views::zip(ranges::views::iota(0), quaternions, normalized_weights)) {
75  all_quaternions.col(i) = normalized_weight * quaternion.coeffs();
76  }
77 
78  const Eigen::Matrix<Scalar, 4, 4> qqt = all_quaternions * all_quaternions.transpose();
79  const Eigen::EigenSolver<Eigen::Matrix<Scalar, 4, 4>> es(qqt);
80 
81  Eigen::Index max_eigenvalue_index;
82  es.eigenvalues().cwiseAbs().maxCoeff(&max_eigenvalue_index);
83  const Eigen::Matrix<std::complex<Scalar>, 4, 1> max_eigenvector = es.eigenvectors().col(max_eigenvalue_index);
84  Eigen::Quaternion<Scalar> quat;
85  // This is not the same as quat{max_eigenvector.real()}. Eigen's internal coefficient order is different from the
86  // constructor one.
87  quat.coeffs() << max_eigenvector.real();
88  return quat;
89 }

105 template <class Range, class WeightsRange, class Scalar>
106 Sophus::Matrix2<Scalar>
107 calculate_covariance(Range&& range, WeightsRange&& normalized_weights, const Sophus::Vector2<Scalar>& mean) {
108  auto translations_view = range | ranges::views::common;
109  auto normalized_weights_view = normalized_weights | ranges::views::common;
110 
111  auto calculate_weighted_sample_covariance = [mean](const auto& value, const auto& weight) {
112  const auto centered = value - mean;
113  return Sophus::Vector3<Scalar>{
114  weight * centered.x() * centered.x(), // weighted sample x autocovariance
115  weight * centered.x() * centered.y(), // weighted sample xy cross-covariance
116  weight * centered.y() * centered.y(), // weighted sample y autocovariance
117  };
118  };
119 
120  // calculate the averaging factor for the weighted covariance estimate
121  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
122  const auto squared_weight_sum = std::transform_reduce(
123  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
124  [](const auto& weight) { return (weight * weight); });
125 
126  // calculate the x autocovariance, xy cross-covariance, and y autocovariance
127  Sophus::Vector3<Scalar> coefficients =
128  std::transform_reduce(
129  translations_view.begin(), translations_view.end(), normalized_weights_view.begin(),
130  Sophus::Vector3<Scalar>::Zero().eval(), std::plus{}, calculate_weighted_sample_covariance) /
131  (1.0 - squared_weight_sum);
132 
133  // create the symmetric 2x2 translation covariance matrix from the coefficients
134  auto covariance_matrix = Sophus::Matrix2<Scalar>{};
135  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
136  return covariance_matrix;
137 }

182 template <
183  class Poses,
184  class Weights,
185  class Pose = ranges::range_value_t<Poses>,
186  class Scalar = typename Pose::Scalar,
187  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE3<Scalar>>>>
188 std::pair<Sophus::SE3<Scalar>, Sophus::Matrix6<Scalar>> estimate(const Poses& poses, const Weights& weights) {
189  assert(std::size(poses) == std::size(weights));
190  assert(std::size(poses) > 0);
191  auto poses_view = poses | ranges::views::common;
192  auto weights_view = weights | ranges::views::common;
193 
194  auto quaternion_view = poses | ranges::views::transform([](const Pose& pose) { return pose.unit_quaternion(); });
195 
196  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
197  auto normalized_weights_view =
198  weights_view | //
199  ranges::views::transform([weights_sum](const auto weight) { return weight / weights_sum; });
200 
201  const auto average_quat = detail::weighted_average_quaternion(quaternion_view, normalized_weights_view);
202 
203  const Sophus::Vector3<Scalar> mean_translation = std::transform_reduce(
204  poses_view.begin(), //
205  poses_view.end(), //
206  normalized_weights_view.begin(), //
207  Sophus::Vector3<Scalar>::Zero().eval(), //
208  std::plus{}, //
209  [](const Pose& pose, const auto& weight) { return pose.translation() * weight; });
210 
211  const Sophus::SE3<Scalar> mean{Sophus::SO3<Scalar>{average_quat}, mean_translation};
212 
213  // calculate the averaging factor for the weighted covariance estimate
214  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
215  const auto squared_weight_sum = std::transform_reduce(
216  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
217  [](const auto& weight) { return (weight * weight); });
218 
219  const Eigen::Matrix<Scalar, 6, 6> covariance =
220  std::transform_reduce(
221  poses_view.begin(), //
222  poses_view.end(), //
223  normalized_weights_view.begin(), //
224  Eigen::Matrix<Scalar, 6, 6>::Zero().eval(), //
225  std::plus{}, //
226  [inverse_mean = mean.inverse()](const Pose& pose, const auto weight) {
227  // Compute deviation from mean in Lie algebra (logarithm of SE3)
228  const Pose delta = inverse_mean * pose;
229  // Accumulate weighted covariance
230  return Eigen::Matrix<Scalar, 6, 6>{weight * (delta.log() * delta.log().transpose())};
231  }) /
232  (1. - squared_weight_sum);
233 
234  return std::pair{mean, covariance};
235 }

255 template <
256  class Poses,
257  class Weights,
258  class Pose = ranges::range_value_t<Poses>,
259  class Scalar = typename Pose::Scalar,
260  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
261 std::pair<Sophus::SE2<Scalar>, Sophus::Matrix3<Scalar>> estimate(Poses&& poses, Weights&& weights) {
262  auto translation_view = poses | ranges::views::transform([](const auto& pose) { return pose.translation(); });
263  auto poses_view = poses | ranges::views::common;
264  auto weights_view = weights | ranges::views::common;
265  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
266  auto normalized_weights_view =
267  weights_view | ranges::views::transform([weights_sum](const auto& weight) { return weight / weights_sum; });
268 
269  // map sophus pose 2D pose into a 4D Eigen vector. Mapping exploits that Sophus stores the 2D transform
270  // as two elements for the linear translation, and two more for the orientation (in complex number form)
271  const auto pose_to_weighted_eigen_vector = [](const auto& pose, const auto& weight) {
272  return Eigen::Map<const Sophus::Vector4<Scalar>>{pose.data()} * weight;
273  };
274 
275  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
276  // that after averaging the complex representation of the orientation the resulting complex is not on the unit circle.
277  // This is expected and the value will be renormalized after having used the non-normal result to estimate the
278  // orientation autocovariance.
279  const Sophus::Vector4<Scalar> mean_pose_vector = std::transform_reduce(
280  poses_view.begin(), poses_view.end(), normalized_weights_view.begin(), Sophus::Vector4<Scalar>::Zero().eval(),
281  std::plus{}, pose_to_weighted_eigen_vector);
282 
283  // Calculate the weighted pose estimation
284  Sophus::SE2<Scalar> estimated_pose = Eigen::Map<const Sophus::SE2<Scalar>>{mean_pose_vector.data()};
285 
286  Sophus::Matrix3<Scalar> covariance_matrix = Sophus::Matrix3<Scalar>::Zero();
287 
288  // Compute the covariance of the translation part.
289  covariance_matrix.template topLeftCorner<2, 2>() =
290  calculate_covariance(translation_view, normalized_weights_view, estimated_pose.translation());
291 
292  // Compute the orientation variance and re-normalize the rotation component.
293  if (estimated_pose.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
294  // Handle the case where both averages are too close to zero.
295  // Return zero yaw and infinite variance.
296  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
297  estimated_pose.so2() = Sophus::SO2<Scalar>{0.0};
298  } else {
299  // See circular standard deviation in
300  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
301  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(estimated_pose.so2().unit_complex().norm());
302  estimated_pose.so2().normalize();
303  }
304  return std::pair{estimated_pose, covariance_matrix};
305 }

320 template <
321  class Scalars,
322  class Weights,
323  class Scalar = ranges::range_value_t<Scalars>,
324  typename = std::enable_if_t<std::is_arithmetic_v<Scalar>>>
325 std::pair<Scalar, Scalar> estimate(Scalars&& scalars, Weights&& weights) {
326  auto weights_view = weights | ranges::views::common;
327  const auto weights_sum = ranges::accumulate(weights, 0.0, std::plus<>{});
328  auto normalized_weights_view =
329  weights_view | ranges::views::transform([weights_sum](auto weight) { return weight / weights_sum; });
330 
331  const Scalar weighted_mean = std::transform_reduce(
332  scalars.begin(), scalars.end(), normalized_weights_view.begin(), 0.0, std::plus<>{}, std::multiplies<>{});
333 
334  const Scalar weighted_squared_deviations = std::transform_reduce(
335  scalars.begin(), scalars.end(), normalized_weights_view.begin(), 0.0, std::plus<>{},
336  [weighted_mean](const auto& scalar, const auto& weight) {
337  return weight * (scalar - weighted_mean) * (scalar - weighted_mean);
338  });
339 
340  const auto number_of_non_zero_weights =
341  static_cast<Scalar>(ranges::count_if(weights_view, [&](auto weight) { return weight > 0; }));
342 
343  const Scalar weighted_variance =
344  weighted_squared_deviations * number_of_non_zero_weights / (number_of_non_zero_weights - 1);
345 
346  return std::pair{weighted_mean, weighted_variance};
347 }
