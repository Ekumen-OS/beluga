107  // that after averaging the complex representation of the orientation the resulting complex is not on the unit

110  std::forward<Values>(values), std::forward<Weights>(normalized_weights),

112  return Eigen::Map<const Sophus::Vector4<Scalar>>{projection(value).data()};
113  });
114 
115  Eigen::Map<Sophus::SE2<Scalar>> mean{mean_vector.data()};
116  mean.so2().normalize();

127  auto operator()(Values&& values, Weights&& normalized_weights, Projection projection = {}) const
128  -> Eigen::Quaternion<Scalar> {
129  static_assert(ranges::input_range<Values>);
130  static_assert(ranges::sized_range<Values>);
131 
132  static_assert(ranges::input_range<Weights>);
133  static_assert(ranges::sized_range<Weights>);
134 
135  // This implementation is based on Sophus' average methods with the variant of non-uniform weights.
136  // See https://github.com/strasdat/Sophus/blob/d0b7315a0d90fc6143defa54596a3a95d9fa10ec/sophus/average.hpp#L135
137  // See https://ntrs.nasa.gov/api/citations/20070017872/downloads/20070017872.pdf equation (13).
138 
139  const auto size = static_cast<int>(ranges::size(values));
140  auto matrix = Eigen::Matrix<Scalar, 4, Eigen::Dynamic>(4, size);
141 
142  auto weights_it = ranges::begin(normalized_weights);
143  auto it = ranges::begin(values);
144 
145  assert(it != ranges::end(values));
146  assert(weights_it != ranges::end(normalized_weights));
147 
148  for (int index = 0; index < size; ++index, ++weights_it, ++it) {
149  matrix.col(index) = *weights_it * projection(*it).coeffs();
150  }
151 
152  assert(it == ranges::end(values));
153  assert(weights_it == ranges::end(normalized_weights));
154 
155  const auto solver = Eigen::SelfAdjointEigenSolver<Eigen::Matrix<Scalar, 4, 4>>{matrix * matrix.transpose()};
156  assert(solver.info() == Eigen::Success);
157 
158  // This is not the same as `result{solver.eigenvectors().col(3).real()}`.
159  // Eigen's internal coefficient order is different from the constructor one.
160  // Eigenvalues are sorted in increasing order, so eigenvalue number 3 is the max.
161  Eigen::Quaternion<Scalar> result;
162  result.coeffs() << solver.eigenvectors().col(3).real();
163  return result;
164  }

230  template <
231  class Values,
232  class Weights,
233  class Projection = ranges::identity,
234  class Value = std::decay_t<std::invoke_result_t<Projection, ranges::range_value_t<Values>>>,
235  std::enable_if_t<std::is_base_of_v<Eigen::MatrixBase<Value>, Value>, int> = 0>
236  auto operator()(
237  Values&& values,
238  Weights&& normalized_weights,
239  const typename Value::PlainMatrix& mean,
240  Projection projection = {}) const {
241  static_assert(ranges::input_range<Values>);
242  static_assert(ranges::input_range<Weights>);
243 
244  constexpr int M = Value::RowsAtCompileTime;
245  constexpr int N = Value::ColsAtCompileTime;
246  static_assert(N == 1);
247  using Scalar = typename Value::Scalar;
248  auto accumulator = Eigen::Matrix<Scalar, M, M>::Zero().eval();
249  auto squared_weight_sum = Scalar{0.0};
250 
251  auto it = ranges::begin(values);
252  const auto last = ranges::end(values);
253  auto weights_it = ranges::begin(normalized_weights);
254 
255  assert(it != last);
256  assert(weights_it != ranges::end(normalized_weights));
257 
258  for (; it != last; ++weights_it, ++it) {
259  const auto& value = projection(*it);
260  const auto weight = *weights_it;
261  const auto centered = value - mean;
262  accumulator.noalias() += weight * centered * centered.transpose();
263  squared_weight_sum += weight * weight;
264  }
265 
266  assert(weights_it == ranges::end(normalized_weights));
267  assert(squared_weight_sum < 1.0);
268 
269  accumulator /= (1.0 - squared_weight_sum); // apply the correction factor to yield an unbiased estimator
270  return accumulator;
271  }

309  class Values,
310  class Weights,
311  class Projection = ranges::identity,
312  class Value = std::decay_t<std::invoke_result_t<Projection, ranges::range_value_t<Values>>>,
313  std::enable_if_t<
314  std::disjunction_v<
315  std::is_base_of<Sophus::SE2Base<Value>, Value>,
316  std::is_base_of<Sophus::SE3Base<Value>, Value>>,
317  int> = 0>
318  auto operator()(Values&& values, Weights&& normalized_weights, const Value& mean, Projection projection = {}) const {
319  static_assert(ranges::input_range<Values>);
320  static_assert(ranges::input_range<Weights>);
321 
322  // For SE3 (and SE2) estimates, we represent the estimate as a noiseless pose and covariance in se3,
323  // the tangent space of the SE3 manifold.
324  //
325  // Users may perform the appropriate conversions to get the covariance matrix into their parametrization of
326  // interest.
327  //
328  // This reasoning is based on "Characterizing the Uncertainty of Jointly Distributed Poses in the Lie
329  // Algebra" by Mangelson et al. (https://robots.et.byu.edu/jmangelson/pubs/2020/mangelson2020tro.pdf).
330  //
331  // See section III. E) "Defining Random Variables over Poses" for more context.
332  auto accumulator = Value::Adjoint::Zero().eval();
333  auto squared_weight_sum = typename Value::Scalar{0.0};
334 
335  auto it = ranges::begin(values);
336  const auto last = ranges::end(values);
337  auto weights_it = ranges::begin(normalized_weights);
338 
339  assert(it != last);
340  assert(weights_it != ranges::end(normalized_weights));
341 
342  const auto inverse_mean = mean.inverse();
343 
344  for (; it != last; ++weights_it, ++it) {
345  const auto& value = projection(*it);
346  const auto weight = *weights_it;
347  const auto centered = (inverse_mean * value).log();
348  accumulator.noalias() += weight * centered * centered.transpose();
349  squared_weight_sum += weight * weight;
350  }
351 
352  assert(weights_it == ranges::end(normalized_weights));
353  assert(squared_weight_sum < 1.0);
354 
355  accumulator /= (1.0 - squared_weight_sum); // apply the correction factor to yield an unbiased estimator
356  return accumulator;
357  }

407 struct estimate_fn {
408  template <
409  class Values,
410  class Weights,
411  class Value = std::decay_t<ranges::range_value_t<Values>>,
412  class = std::enable_if_t<std::disjunction_v<
413  std::is_floating_point<Value>,
414  std::is_base_of<Eigen::MatrixBase<Value>, Value>,
415  std::is_base_of<Sophus::SE3Base<Value>, Value>>>>
416  auto operator()(Values&& values, Weights&& weights) const {
417  static_assert(ranges::forward_range<Values>); // must allow multi-pass
418  static_assert(ranges::forward_range<Weights>); // must allow multi-pass
419 
420  auto normalized_weights = weights | ranges::views::transform([sum = ranges::accumulate(weights, 0.0)](auto weight) {
421  return weight / sum;
422  });
423 
424  const auto mean = beluga::mean(values, normalized_weights);
425  const auto variance = beluga::covariance(values, normalized_weights, mean);
426  return std::make_pair(mean, variance);
427  }

429  template <
430  class Values,
431  class Weights,
432  class Value = std::decay_t<ranges::range_value_t<Values>>,
433  class Scalar = typename Value::Scalar,
434  class = std::enable_if_t<std::is_base_of_v<Sophus::SE2Base<Value>, Value>>>
435  auto operator()(Values&& values, Weights&& weights) const -> std::pair<Sophus::SE2<Scalar>, Sophus::Matrix3<Scalar>> {
436  static_assert(ranges::forward_range<Values>); // must allow multi-pass
437  static_assert(ranges::forward_range<Weights>); // must allow multi-pass
438 
439  auto normalized_weights = weights | ranges::views::transform([sum = ranges::accumulate(weights, 0.0)](auto weight) {
440  return weight / sum;
441  });
442 
443  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
444  // that after averaging the complex representation of the orientation the resulting complex is not on the unit
445  // circle. This is expected and the value will be renormalized after having used the non-normal result to estimate
446  // the orientation autocovariance.
447  const Sophus::Vector4<Scalar> mean_vector = beluga::mean(values, normalized_weights, [](const auto& value) {
448  return Eigen::Map<const Sophus::Vector4<Scalar>>{value.data()};
449  });
450 
451  auto mean = Sophus::SE2<Scalar>{Eigen::Map<const Sophus::SE2<Scalar>>{mean_vector.data()}};
452  auto covariance = Sophus::Matrix3<Scalar>::Zero().eval();
453 
454  // Compute the covariance of the translation part.
455  covariance.template topLeftCorner<2, 2>() = beluga::covariance(
456  values, normalized_weights, mean.translation(), [](const auto& value) { return value.translation(); });
457 
458  // Compute the orientation variance and re-normalize the rotation component (after using the non-normal result).
459  if (mean.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
460  // Handle the case where both averages are too close to zero.
461  // Return zero yaw and infinite variance.
462  covariance.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
463  mean.so2() = Sophus::SO2<Scalar>{0.0};
464  // TODO(nahuel): Consider breaking the existing API and return
465  // an optional to handle degenerate cases just like Sophus does.
466  } else {
467  // See circular standard deviation in
468  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
469  covariance.coeffRef(2, 2) = -2.0 * std::log(mean.so2().unit_complex().norm());
470  mean.so2().normalize();
471  }
472 
473  return std::make_pair(mean, covariance);
474  }
