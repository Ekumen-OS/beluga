51 template <class Range, class WeightsRange, class Scalar>
52 Sophus::Matrix2<Scalar>
53 calculate_covariance(Range&& range, WeightsRange&& normalized_weights, const Sophus::Vector2<Scalar>& mean) {
54  auto translations_view = range | ranges::views::common;
55  auto normalized_weights_view = normalized_weights | ranges::views::common;
56 
57  auto calculate_weighted_sample_covariance = [mean](const auto& value, const auto& weight) {
58  const auto centered = value - mean;
59  return Sophus::Vector3<Scalar>{
60  weight * centered.x() * centered.x(), // weighted sample x autocovariance
61  weight * centered.x() * centered.y(), // weighted sample xy cross-covariance
62  weight * centered.y() * centered.y(), // weighted sample y autocovariance
63  };
64  };
65 
66  // calculate the averaging factor for the weighted covariance estimate
67  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
68  const auto squared_weight_sum = std::transform_reduce(
69  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
70  [](const auto& weight) { return (weight * weight); });
71 
72  // calculate the x autocovariance, xy cross-covariance, and y autocovariance
73  Sophus::Vector3<Scalar> coefficients =
74  std::transform_reduce(
75  translations_view.begin(), translations_view.end(), normalized_weights_view.begin(),
76  Sophus::Vector3<Scalar>::Zero().eval(), std::plus{}, calculate_weighted_sample_covariance) /
77  (1.0 - squared_weight_sum);
78 
79  // create the symmetric 2x2 translation covariance matrix from the coefficients
80  auto covariance_matrix = Sophus::Matrix2<Scalar>{};
81  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
82  return covariance_matrix;
83 }

123 template <
124  class Poses,
125  class Weights,
126  class Pose = ranges::range_value_t<Poses>,
127  class Scalar = typename Pose::Scalar,
128  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
129 std::pair<Sophus::SE2<Scalar>, Sophus::Matrix3<Scalar>> estimate(Poses&& poses, Weights&& weights) {
130  auto translation_view = poses | ranges::views::transform([](const auto& pose) { return pose.translation(); });
131  auto weights_view = weights | ranges::views::common;
132  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
133  auto normalized_weights_view =
134  weights_view | ranges::views::transform([weights_sum](const auto& weight) { return weight / weights_sum; });
135 
136  // map sophus pose 2D pose into a 4D Eigen vector. Mapping exploits that Sophus stores the 2D transform
137  // as two elements for the linear translation, and two more for the orientation (in complex number form)
138  const auto pose_to_weighted_eigen_vector = [](const auto& pose, const auto& weight) {
139  return Eigen::Map<const Sophus::Vector4<Scalar>>{pose.data()} * weight;
140  };
141 
142  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
143  // that after averaging the complex representation of the orientation the resulting complex is not on the unit circle.
144  // This is expected and the value will be renormalized after having used the non-normal result to estimate the
145  // orientation autocovariance.
146  const Sophus::Vector4<Scalar> mean_pose_vector = std::transform_reduce(
147  poses.begin(), poses.end(), normalized_weights_view.begin(), Sophus::Vector4<Scalar>::Zero().eval(), std::plus{},
148  pose_to_weighted_eigen_vector);
149 
150  // Calculate the weighted pose estimation
151  Sophus::SE2<Scalar> estimated_pose = Eigen::Map<const Sophus::SE2<Scalar>>{mean_pose_vector.data()};
152 
153  Sophus::Matrix3<Scalar> covariance_matrix = Sophus::Matrix3<Scalar>::Zero();
154 
155  // Compute the covariance of the translation part.
156  covariance_matrix.template topLeftCorner<2, 2>() =
157  calculate_covariance(translation_view, normalized_weights_view, estimated_pose.translation());
158 
159  // Compute the orientation variance and re-normalize the rotation component.
160  if (estimated_pose.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
161  // Handle the case where both averages are too close to zero.
162  // Return zero yaw and infinite variance.
163  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
164  estimated_pose.so2() = Sophus::SO2<Scalar>{0.0};
165  } else {
166  // See circular standard deviation in
167  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
168  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(estimated_pose.so2().unit_complex().norm());
169  estimated_pose.so2().normalize();
170  }
171  return std::pair{estimated_pose, covariance_matrix};
172 }

192 std::pair<Scalar, Scalar> estimate(Scalars&& scalars, Weights&& weights) {
193  auto weights_view = weights | ranges::views::common;
194  const auto weights_sum = ranges::accumulate(weights, 0.0, std::plus<>{});
195  auto normalized_weights_view =
196  weights_view | ranges::views::transform([weights_sum](auto weight) { return weight / weights_sum; });
197 
198  const Scalar weighted_mean = std::transform_reduce(
199  scalars.begin(), scalars.end(), normalized_weights_view.begin(), 0.0, std::plus<>{}, std::multiplies<>{});
200 
201  const Scalar weighted_squared_deviations = std::transform_reduce(
202  scalars.begin(), scalars.end(), normalized_weights_view.begin(), 0.0, std::plus<>{},
203  [weighted_mean](const auto& scalar, const auto& weight) {
204  return weight * (scalar - weighted_mean) * (scalar - weighted_mean);
205  });
206 
207  const auto number_of_non_zero_weights =
208  static_cast<Scalar>(ranges::count_if(weights_view, [&](auto weight) { return weight > 0; }));
209 
210  const Scalar weighted_sd =
211  std::sqrt(weighted_squared_deviations * number_of_non_zero_weights / (number_of_non_zero_weights - 1));
212 
213  return std::pair{weighted_mean, weighted_sd};
214 }
