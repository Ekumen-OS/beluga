99  [[nodiscard]] auto operator()(Control&& action) const {
100  const auto& [pose, previous_pose] = action;
101 
102  const auto translation = pose.translation() - previous_pose.translation();
103  const double distance = translation.norm();
104  const double distance_variance = distance * distance;
105 
106  const auto& previous_orientation = previous_pose.so2();
107  const auto& current_orientation = pose.so2();
108  const auto heading_rotation = Sophus::SO2d{std::atan2(translation.y(), translation.x())};
109  const auto first_rotation =
110  distance > params_.distance_threshold ? heading_rotation * previous_orientation.inverse() : Sophus::SO2d{};
111  const auto second_rotation = current_orientation * previous_orientation.inverse() * first_rotation.inverse();
112 
113  using DistributionParam = typename std::normal_distribution<double>::param_type;
114  const auto first_rotation_params = DistributionParam{
115  first_rotation.log(), std::sqrt(
116  params_.rotation_noise_from_rotation * rotation_variance(first_rotation) +
117  params_.rotation_noise_from_translation * distance_variance)};
118  const auto translation_params = DistributionParam{
119  distance, std::sqrt(
120  params_.translation_noise_from_translation * distance_variance +
121  params_.translation_noise_from_rotation *
122  (rotation_variance(first_rotation) + rotation_variance(second_rotation)))};
123  const auto second_rotation_params = DistributionParam{
124  second_rotation.log(), std::sqrt(
125  params_.rotation_noise_from_rotation * rotation_variance(second_rotation) +
126  params_.rotation_noise_from_translation * distance_variance)};
127 
128  return [=](const state_type& state, auto& gen) {
129  static thread_local auto distribution = std::normal_distribution<double>{};
130  const auto first_rotation = Sophus::SO2d{distribution(gen, first_rotation_params)};
131  const auto translation = Eigen::Vector2d{distribution(gen, translation_params), 0.0};
132  const auto second_rotation = Sophus::SO2d{distribution(gen, second_rotation_params)};
133  return state * Sophus::SE2d{first_rotation, Eigen::Vector2d{0.0, 0.0}} *
134  Sophus::SE2d{second_rotation, translation};
135  };
136  }

142  // Treat backward and forward motion symmetrically for the noise models.
