126  static ValueGrid2<float> make_likelihood_field(const param_type& params, const OccupancyGrid& grid) {
127  const auto squared_distance = [&grid](std::size_t first, std::size_t second) {
128  return static_cast<float>((grid.coordinates_at(first) - grid.coordinates_at(second)).squaredNorm());
129  };
130 
132  const double two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit;
133  assert(two_squared_sigma > 0.0);
134 
135  const double amplitude = params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi()));
136  assert(amplitude > 0.0);
137 
138  const double offset = params.z_random / params.max_laser_distance;
139 
140  const auto to_likelihood = [amplitude, two_squared_sigma, offset](double squared_distance) {
141  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
142  };
143 
144  const auto neighborhood = [&grid](std::size_t index) { return grid.neighborhood4(index); };
145 
146  const auto squared_max_distance = static_cast<float>(params.max_obstacle_distance * params.max_obstacle_distance);
147 
148  // determine distances to obstacles and calculate likelihood values in-place
149  // to minimize memory usage when dealing with large maps
150  auto distance_map =
151  nearest_obstacle_distance_map(grid.obstacle_mask(), squared_distance, neighborhood, squared_max_distance);
152 
153  if (params.model_unknown_space) {
154  const auto inverse_max_distance = 1 / params.max_laser_distance;
155  const auto squared_background_distance =
156  -two_squared_sigma * std::log((inverse_max_distance - offset) / amplitude);
157 
158  distance_map |= beluga::actions::overlay(
159  grid.unknown_mask(), std::min(squared_max_distance, static_cast<float>(squared_background_distance)));
160  }
161 
162  auto likelihood_values = std::move(distance_map) | //
163  ranges::actions::transform(to_likelihood);
164 
165  return ValueGrid2<float>{std::move(likelihood_values), grid.width(), grid.resolution()};
166  }
