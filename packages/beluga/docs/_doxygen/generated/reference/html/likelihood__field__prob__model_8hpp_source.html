68  [[nodiscard]] auto operator()(measurement_type&& points) const {
69  return [this, points = std::move(points)](const state_type& state) -> weight_type {
70  const auto transform = this->world_to_likelihood_field_transform_ * state;
71  const auto x_offset = transform.translation().x();
72  const auto y_offset = transform.translation().y();
73  const auto cos_theta = transform.so2().unit_complex().x();
74  const auto sin_theta = transform.so2().unit_complex().y();
75  const auto unknown_space_occupancy_prob = static_cast<float>(1. / this->params_.max_laser_distance);
76 
77  return std::exp(std::transform_reduce(
78  points.cbegin(), points.cend(), 0.0, std::plus{},
79  [this, x_offset, y_offset, cos_theta, sin_theta, unknown_space_occupancy_prob](const auto& point) {
80  // Transform the end point of the laser to the grid local coordinate system.
81  // Not using Eigen/Sophus because they make the routine x10 slower.
82  // See `benchmark_likelihood_field_model.cpp` for reference.
83  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
84  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
85  const auto pz =
86  static_cast<double>(this->likelihood_field_.data_near(x, y).value_or(unknown_space_occupancy_prob));
87  return std::log(pz);
