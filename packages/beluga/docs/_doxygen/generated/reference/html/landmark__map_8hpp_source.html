117  [[nodiscard]] std::optional<LandmarkBearing3> find_closest_bearing_landmark(
118  const LandmarkBearing3& detection_bearing_in_sensor,
119  const LandmarkCategory& detection_category,
120  const world_pose_type& sensor_pose_in_world) const {
121  // only consider those that have the same detection id (category)
122  auto same_category_landmarks_view =
123  landmarks_ | ranges::views::filter([detection_category = detection_category](const auto& l) {
124  return detection_category == l.category;
125  });
126 
127  // find the landmark that minimizes the bearing error
128  const auto world_in_sensor_transform = sensor_pose_in_world.inverse();
129 
130  // This whole search thing is very expensive, with objects getting created, normalized and
131  // destroyed multiple times and the same transformations being calculated over and over again.
132  // This will only work as a proof-of-concept, but it needs to be optimized for large numbers of
133  // landmarks.
134  const auto minimization_function = [&detection_bearing_in_sensor, &world_in_sensor_transform](
135  const auto& a, const auto& b) {
136  const auto& landmark_a_position_in_world = a.detection_position_in_robot;
137  const auto& landmark_b_position_in_world = b.detection_position_in_robot;
138 
139  // convert the landmark locations relative to the sensor frame
140  const auto landmark_a_bearing_in_sensor = (world_in_sensor_transform * landmark_a_position_in_world).normalized();
141  const auto landmark_b_bearing_in_sensor = (world_in_sensor_transform * landmark_b_position_in_world).normalized();
142 
143  // find the landmark that minimizes the bearing error by maximizing the dot product against the
144  // detection bearing vector
145  const auto dot_product_a = landmark_a_bearing_in_sensor.dot(detection_bearing_in_sensor);
146  const auto dot_product_b = landmark_b_bearing_in_sensor.dot(detection_bearing_in_sensor);
147 
148  return dot_product_a > dot_product_b;
149  };
150 
151  auto min = std::min_element(
152  same_category_landmarks_view.begin(), same_category_landmarks_view.end(), minimization_function);
153 
154  if (min == same_category_landmarks_view.end()) {
155  return std::nullopt;
156  }
157 
158  // find the normalized bearing vector to the landmark, relative to the sensor frame
159  const auto landmark_position_in_sensor = world_in_sensor_transform * min->detection_position_in_robot;
160  return landmark_position_in_sensor.normalized();
161  }
