56 inline beluga_ros::msg::Pose& toMsg(const Sophus::SE2<Scalar>& in, beluga_ros::msg::Pose& out) {
57  const double theta = in.so2().log();
58  out.position.x = in.translation().x();
59  out.position.y = in.translation().y();
