406 struct RepresentativeBinHash {
408  std::size_t operator()(const Sophus::SE2d& s) const noexcept {
409  std::size_t h1 = std::hash<double>{}(s.translation().x());
410  std::size_t h2 = std::hash<double>{}(s.translation().y());
411  std::size_t h3 = std::hash<double>{}(s.so2().log());
412  return h1 ^ (h2 << 1) ^ (h3 << 2);
413  }
414 };

417 struct RepresentativeBinEqual {
419  bool operator()(const Sophus::SE2d& lhs, const Sophus::SE2d& rhs) const noexcept {
420  // good enough, since copies of the same candidate are expected to be identical copies
421  return lhs.translation().x() == rhs.translation().x() && //
422  lhs.translation().y() == rhs.translation().y() && //
423  lhs.so2().log() == rhs.so2().log();
424  }

427 // Particle weights from the filter may or may not be representative of the
428 // true distribution. If we resampled, they are not, and there will be multiple copies
429 // of the most likely candidates, all with unit weight. In this case the number of copies
430 // is a proxy for the prob density at each candidate. If we did not resample before updating
431 // the estimation and publishing this message (which can happen if the resample interval
432 // is set to something other than 1), then all particles are expected to be different
433 // and their weights are proportional to the prob density at each candidate.
434 //
435 // Only the combination of both the state distribution and the candidate weights together
436 // provide information about the probability density at each candidate.
437 // To handle both cases, we group repeated candidates and compute the accumulated weight.
438 template <class ParticleFilter>
439 nav2_msgs::msg::ParticleCloud make_representative_particle_cloud(
440  const ParticleFilter& particle_filter,
441  std::string_view global_frame,
442  rclcpp::Time now) {
443  std::unordered_map<Sophus::SE2d, RepresentativeData, RepresentativeBinHash, RepresentativeBinEqual>
444  representatives_map;
445  representatives_map.reserve(particle_filter.particles().size());
446 
447  double max_weight = 1e-5; // never risk dividing by zero
448 
449  for (const auto& [state, weight] : particle_filter.particles()) {
450  auto& representative = representatives_map[state]; // if the element does not exist, create it
451  representative.state = state;
452  representative.weight += weight;
453  if (representative.weight > max_weight) {
454  max_weight = representative.weight;
455  }
456  }
457 
458  auto message = nav2_msgs::msg::ParticleCloud{};
459  message.header.stamp = now;
460  message.header.frame_id = global_frame;
461  message.particles.reserve(particle_filter.particles().size());
462 
463  for (const auto& [key, representative] : representatives_map) {
464  auto& particle = message.particles.emplace_back();
465  tf2::toMsg(representative.state, particle.pose);
466  particle.weight = representative.weight / max_weight;
467  }
468 
469  return message;
470 }
