51 template <class Range, class DistanceFunction, class NeighborsFunction>
52 auto nearest_obstacle_distance_map(
53  Range&& obstacle_map,
54  DistanceFunction&& distance_function,
55  NeighborsFunction&& neighbors_function) {
56  struct IndexPair {
57  std::size_t nearest_obstacle_index;
58  std::size_t index;
59  };
60 
61  using DistanceType = std::invoke_result_t<DistanceFunction, std::size_t, std::size_t>;
62  auto distance_map = std::vector<DistanceType>(ranges::size(obstacle_map));
63  auto visited = std::vector<bool>(ranges::size(obstacle_map), false);
64 
65  auto compare = [&distance_map](const IndexPair& first, const IndexPair& second) {
66  return distance_map[first.index] > distance_map[second.index];
67  };
68  auto queue = std::priority_queue<IndexPair, std::vector<IndexPair>, decltype(compare)>{compare};
69 
70  for (auto [index, is_obstacle] : ranges::views::enumerate(obstacle_map)) {
71  if (is_obstacle) {
72  visited[index] = true;
73  distance_map[index] = 0;
74  queue.push(IndexPair{index, index}); // The nearest obstacle is itself
75  }
76  }
77 
78  while (!queue.empty()) {
79  auto parent = queue.top();
80  queue.pop();
81  for (std::size_t index : neighbors_function(parent.index)) {
82  if (!visited[index]) {
83  visited[index] = true;
84  distance_map[index] = distance_function(parent.nearest_obstacle_index, index);
85  queue.push(IndexPair{parent.nearest_obstacle_index, index});
86  }
87  }
88  }
89 
90  return distance_map;
91 }
