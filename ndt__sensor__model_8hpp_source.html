41 template <int N>
42 struct CellHasher {
44  size_t operator()(const Eigen::Matrix<int, N, 1> vector) const {
45  size_t seed = 0;
46  for (auto i = 0L; i < vector.size(); ++i) {
47  auto elem = *(vector.data() + i);
48  seed ^= std::hash<int>()(elem) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
49  }
50  return seed;
51  }
52 };

55 template <int NDim, typename Scalar = double>
56 inline NDTCell<NDim, Scalar> fit_points(const std::vector<Eigen::Matrix<Scalar, NDim, 1>>& points) {
57  static constexpr double kMinVariance = 1e-5;
58  Eigen::Map<const Eigen::Matrix<Scalar, NDim, Eigen::Dynamic>> points_view(
59  reinterpret_cast<const Scalar*>(points.data()), 2, static_cast<int64_t>(points.size()));
60  const Eigen::Matrix<Scalar, NDim, 1> mean = points_view.rowwise().mean();
61  const Eigen::Matrix<Scalar, NDim, Eigen::Dynamic> centered = points_view.colwise() - mean;
62  // Use sample covariance.
63  Eigen::Matrix<Scalar, NDim, Eigen::Dynamic> cov =
64  (centered * centered.transpose()) / static_cast<double>(points_view.cols() - 1);
65  cov(0, 0) = std::max(cov(0, 0), kMinVariance);
66  cov(1, 1) = std::max(cov(1, 1), kMinVariance);
67  if constexpr (NDim == 3) {
68  cov(2, 2) = std::max(cov(2, 2), kMinVariance);
69  }
70  return NDTCell<NDim, Scalar>{mean, cov};
71 }

76 template <int NDim, typename Scalar = double>
77 inline std::vector<NDTCell<NDim, Scalar>> to_cells(
78  const std::vector<Eigen::Matrix<Scalar, NDim, 1>>& points,
79  const double resolution) {
80  static constexpr int kMinPointsPerCell = 5;
81 
82  Eigen::Map<const Eigen::Matrix<Scalar, NDim, Eigen::Dynamic>> points_view(
83  reinterpret_cast<const Scalar*>(points.data()), NDim, static_cast<int64_t>(points.size()));
84 
85  std::vector<NDTCell<NDim, Scalar>> ret;
86  ret.reserve(static_cast<size_t>(points_view.cols()) / kMinPointsPerCell);
87 
88  std::unordered_map<Eigen::Matrix<int, NDim, 1>, std::vector<Eigen::Matrix<Scalar, NDim, 1>>, CellHasher<NDim>>
89  cell_grid;
90  for (const Eigen::Matrix<Scalar, NDim, 1>& col : points) {
91  cell_grid[(col / resolution).template cast<int>()].emplace_back(col);
92  }
93 
94  for (const auto& [cell, points_in_cell] : cell_grid) {
95  if (points_in_cell.size() < kMinPointsPerCell) {
96  continue;
97  }
98  ret.push_back(fit_points<NDim, Scalar>(points_in_cell));
99  }
100 
101  return ret;
102 }

122 template <typename SparseGridT>
123 class NDTSensorModel {
124  public:
125  static_assert(std::is_same_v<SparseValueGrid<typename SparseGridT::map_type>, SparseGridT>);
127  using ndt_cell_type = typename SparseGridT::mapped_type;
128  static_assert(ndt_cell_type::num_dim == 2, "NDT sensor model is only implemented for 2D problems.");
130  using state_type = std::conditional_t<
131  ndt_cell_type::num_dim == 2,
132  Sophus::SE2<typename ndt_cell_type::scalar_type>,
133  Sophus::SE3<typename ndt_cell_type::scalar_type>>;
135  using weight_type = double;
137  using measurement_type = std::vector<Eigen::Matrix<typename ndt_cell_type::scalar_type, ndt_cell_type::num_dim, 1>>;
139  using map_type = SparseGridT;
141  using param_type = NDTModelParam;
142 
144 
150  NDTSensorModel(param_type params, SparseGridT cells_data)
151  : params_{std::move(params)}, cells_data_{std::move(cells_data)} {}
152 
154 
159  [[nodiscard]] auto operator()(measurement_type&& points) const {
160  return [this, cells = detail::to_cells<ndt_cell_type::num_dim, typename ndt_cell_type::scalar_type>(
161  points, cells_data_.resolution())](const state_type& state) -> weight_type {
162  return std::transform_reduce(
163  cells.cbegin(), cells.cend(), 1.0, std::plus{},
164  [this, state](const ndt_cell_type& ndt_cell) { return likelihood_at(state * ndt_cell); });
165  };
166  }
167 
170  [[nodiscard]] double likelihood_at(const ndt_cell_type& measurement) const {
171  const auto maybe_cell = cells_data_.data_near(measurement.mean);
172  if (!maybe_cell.has_value()) {
173  return params_.minimum_likelihood;
174  }
175  return maybe_cell->likelihood_at(measurement, params_.d1, params_.d2);
176  }
177 
178  private:
179  const param_type params_;
180  const SparseGridT cells_data_;
181 };

195 template <typename NDTMapRepresentationT>
196 NDTMapRepresentationT load_from_hdf5_2d(const std::filesystem::path& path_to_hdf5_file) {
197  static_assert(std::is_same_v<typename NDTMapRepresentationT::mapped_type, NDTCell2d>);
198  static_assert(std::is_same_v<typename NDTMapRepresentationT::key_type, Eigen::Vector2i>);
199  if (!std::filesystem::exists(path_to_hdf5_file) || std::filesystem::is_directory(path_to_hdf5_file)) {
200  std::stringstream ss;
201  ss << "Couldn't find a valid HDF5 file at " << path_to_hdf5_file;
202  throw std::invalid_argument(ss.str());
203  }
204 
205  H5::H5File file(path_to_hdf5_file, H5F_ACC_RDONLY);
206  H5::DataSet means_dataset = file.openDataSet("means");
207  H5::DataSet covariances_dataset = file.openDataSet("covariances");
208  H5::DataSet resolution_dataset = file.openDataSet("resolution");
209  H5::DataSet cells_dataset = file.openDataSet("cells");
210 
211  std::array<hsize_t, 2> dims_out;
212  means_dataset.getSpace().getSimpleExtentDims(dims_out.data(), nullptr);
213 
214  Eigen::Matrix2Xd means_matrix(dims_out[1], dims_out[0]);
215  Eigen::Matrix2Xi cells_matrix(dims_out[1], dims_out[0]);
216 
217  means_dataset.read(means_matrix.data(), H5::PredType::NATIVE_DOUBLE);
218  cells_dataset.read(cells_matrix.data(), H5::PredType::NATIVE_INT);
219 
220  std::vector<Eigen::Array<double, 2, 2>> covariances(dims_out[0]);
221  covariances_dataset.read(covariances.data(), H5::PredType::NATIVE_DOUBLE);
222 
223  double resolution;
224 
225  resolution_dataset.read(&resolution, H5::PredType::NATIVE_DOUBLE);
226 
227  typename NDTMapRepresentationT::map_type map{};
228 
229  // Note: Ranges::zip_view doesn't seem to work in old Eigen.
230  for (size_t i = 0; i < covariances.size(); ++i) {
231  map[cells_matrix.col(static_cast<Eigen::Index>(i))] =
232  NDTCell2d{means_matrix.col(static_cast<Eigen::Index>(i)), covariances.at(i)};
233  }
234 
235  return NDTMapRepresentationT{std::move(map), resolution};
236 }
