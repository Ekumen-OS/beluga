44 struct assign_fn {
46  template <
47  class Range,
48  class Fn,
49  std::enable_if_t<ranges::range<Range>, int> = 0,
50  std::enable_if_t<is_range_closure_v<Fn>, int> = 0>
51  constexpr auto operator()(Range& range, Fn fn) const -> Range& {
52  auto&& view = fn(range);
53  if constexpr (!std::is_same_v<Range, std::decay_t<decltype(view)>>) {
54  // If the result of invoking the closure is not the range itself,
55  // then we need to convert the view and assign it to the input range.
56  range = std::move(view) | ranges::to<Range>;
57  }
58  return range;
59  }
