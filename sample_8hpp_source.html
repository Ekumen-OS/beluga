153  template <class Range, class URNG>
154  constexpr auto sample_from_range(Range&& range, URNG& engine) const {
155  static_assert(ranges::sized_range<Range>);
156  static_assert(ranges::random_access_range<Range>);
157  if constexpr (beluga::is_particle_range_v<Range>) {
158  return sample_from_range(beluga::views::states(range), beluga::views::weights(range), engine) |
159  ranges::views::transform(beluga::make_from_state<ranges::range_value_t<Range>>);
160  } else {
161  using result_type = ranges::range_difference_t<Range>;
162  auto distribution =
163  std::uniform_int_distribution<result_type>{0, static_cast<result_type>(ranges::size(range) - 1)};
164  return sample_view{ranges::views::all(std::forward<Range>(range)), std::move(distribution), engine};
165  }

187  template <class T, class U>
188  constexpr auto operator()(T&& t, U&& u) const {
189  if constexpr (ranges::range<T> && ranges::range<U>) {
190  auto& engine = ranges::detail::get_random_engine();
191  return sample_from_range(std::forward<T>(t), std::forward<U>(u), engine);
192  } else if constexpr (is_random_distribution_v<T>) {
193  static_assert(std::is_lvalue_reference_v<U&&>); // Assume U is a URNG
194  return sample_from_distribution(std::forward<T>(t), u);

196  static_assert(ranges::range<T>);

205  if constexpr (ranges::range<T>) {
