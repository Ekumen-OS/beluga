76  [[nodiscard]] Eigen::Vector2i cell_near(double x, double y) const {
77  const auto inv_resolution = 1. / this->self().resolution();
78  const auto scaled_x = x * inv_resolution;
79  const auto scaled_y = y * inv_resolution;
80  // Use a poor man's replacement of `std::floor` because

83  const auto xi = static_cast<int>(scaled_x) - (scaled_x < 0.0);
84  const auto yi = static_cast<int>(scaled_y) - (scaled_y < 0.0);

86  }
