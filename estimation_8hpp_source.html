66 template <class Range, class WeightsRange, class Scalar>
67 Sophus::Matrix2<Scalar>
68 calculate_covariance(Range&& range, WeightsRange&& normalized_weights, const Sophus::Vector2<Scalar>& mean) {
69  auto translations_view = range | ranges::views::common;
70  auto normalized_weights_view = normalized_weights | ranges::views::common;
71 
72  auto calculate_weighted_sample_covariance = [mean](const auto& value, const auto& weight) {
73  const auto centered = value - mean;
74  return Sophus::Vector3<Scalar>{
75  weight * centered.x() * centered.x(), // weighted sample x autocovariance
76  weight * centered.x() * centered.y(), // weighted sample xy cross-covariance
77  weight * centered.y() * centered.y(), // weighted sample y autocovariance
78  };
79  };
80 
81  // calculate the averaging factor for the weighted covariance estimate
82  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
83  const auto squared_weight_sum = std::transform_reduce(
84  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
85  [](const auto& weight) { return (weight * weight); });
86 
87  // calculate the x autocovariance, xy cross-covariance, and y autocovariance
88  Sophus::Vector3<Scalar> coefficients =
89  std::transform_reduce(
90  translations_view.begin(), translations_view.end(), normalized_weights_view.begin(),
91  Sophus::Vector3<Scalar>::Zero().eval(), std::plus{}, calculate_weighted_sample_covariance) /
92  (1.0 - squared_weight_sum);
93 
94  // create the symmetric 2x2 translation covariance matrix from the coefficients
95  auto covariance_matrix = Sophus::Matrix2<Scalar>{};
96  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
97  return covariance_matrix;
98 }

111 template <class Range, class Scalar>
112 Sophus::Matrix2<Scalar> calculate_covariance(Range&& range, const Sophus::Vector2<Scalar>& mean) {
113  const auto sample_count = range.size();
114  return calculate_covariance(
115  range,
116  ranges::views::repeat_n(1.0 / static_cast<Scalar>(sample_count), static_cast<std::ptrdiff_t>(sample_count)),
117  mean);

138 template <
139  class Poses,
140  class Weights,
141  class Pose = ranges::range_value_t<Poses>,
142  class Scalar = typename Pose::Scalar,
143  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
144 std::pair<Sophus::SE2<Scalar>, Sophus::Matrix3<Scalar>> estimate(Poses&& poses, Weights&& weights) {
145  auto translation_view = poses | ranges::views::transform([](const auto& pose) { return pose.translation(); });
146  auto weights_view = weights | ranges::views::common;
147  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
148  auto normalized_weights_view =
149  weights_view | ranges::views::transform([weights_sum](const auto& weight) { return weight / weights_sum; });
150 
151  // map sophus pose 2D pose into a 4D Eigen vector. Mapping exploits that Sophus stores the 2D transform
152  // as two elements for the linear translation, and two more for the orientation (in complex number form)
153  const auto pose_to_weighted_eigen_vector = [](const auto& pose, const auto& weight) {
154  return Eigen::Map<const Sophus::Vector4<Scalar>>{pose.data()} * weight;
155  };
156 
157  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
158  // that after averaging the complex representation of the orientation the resulting complex is not on the unit circle.
159  // This is expected and the value will be renormalized after having used the non-normal result to estimate the
160  // orientation autocovariance.
161  const Sophus::Vector4<Scalar> mean_pose_vector = std::transform_reduce(
162  poses.begin(), poses.end(), normalized_weights_view.begin(), Sophus::Vector4<Scalar>::Zero().eval(), std::plus{},
163  pose_to_weighted_eigen_vector);
164 
165  // Calculate the weighted pose estimation
166  Sophus::SE2<Scalar> estimated_pose = Eigen::Map<const Sophus::SE2<Scalar>>{mean_pose_vector.data()};
167 
168  Sophus::Matrix3<Scalar> covariance_matrix = Sophus::Matrix3<Scalar>::Zero();
169 
170  // Compute the covariance of the translation part.
171  covariance_matrix.template topLeftCorner<2, 2>() =
172  calculate_covariance(translation_view, normalized_weights_view, estimated_pose.translation());
173 
174  // Compute the orientation variance and re-normalize the rotation component.
175  if (estimated_pose.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
176  // Handle the case where both averages are too close to zero.
177  // Return zero yaw and infinite variance.
178  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
179  estimated_pose.so2() = Sophus::SO2<Scalar>{0.0};
180  } else {
181  // See circular standard deviation in
182  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
183  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(estimated_pose.so2().unit_complex().norm());
184  estimated_pose.so2().normalize();
185  }
186  return std::pair{estimated_pose, covariance_matrix};
187 }
