65 template <class Range, class WeightsRange, class Scalar>
66 Eigen::Matrix2<Scalar>
67 calculate_covariance(Range&& range, WeightsRange&& normalized_weights, const Eigen::Vector2<Scalar>& mean) {
68  auto translations_view = range | ranges::views::common;
69  auto normalized_weights_view = normalized_weights | ranges::views::common;
70 
71  auto calculate_weighted_sample_covariance = [mean](const auto& value, const auto& weight) {
72  const auto centered = value - mean;
73  return Eigen::Vector3<Scalar>{
74  weight * centered.x() * centered.x(), // weighted sample x autocovariance
75  weight * centered.x() * centered.y(), // weighted sample xy cross-covariance
76  weight * centered.y() * centered.y(), // weighted sample y autocovariance
77  };
78  };
79 
80  // calculate the averaging factor for the weighted covariance estimate
81  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
82  const auto squared_weight_sum = std::transform_reduce(
83  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
84  [](const auto& weight) { return (weight * weight); });
85 
86  // calculate the x autocovariance, xy cross-covariance, and y autocovariance
87  Eigen::Vector3<Scalar> coefficients =
88  std::transform_reduce(
89  translations_view.begin(), translations_view.end(), normalized_weights_view.begin(),
90  Eigen::Vector3<Scalar>::Zero().eval(), std::plus{}, calculate_weighted_sample_covariance) /
91  (1.0 - squared_weight_sum);
92 
93  // create the symmetric 2x2 translation covariance matrix from the coefficients
94  auto covariance_matrix = Eigen::Matrix2<Scalar>{};
95  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
96  return covariance_matrix;
97 }

110 template <class Range, class Scalar>
111 Eigen::Matrix2<Scalar> calculate_covariance(Range&& range, const Eigen::Vector2<Scalar>& mean) {
112  const auto sample_count = range.size();
113  return calculate_covariance(
114  range,
115  ranges::views::repeat_n(1.0 / static_cast<Scalar>(sample_count), static_cast<std::ptrdiff_t>(sample_count)),
116  mean);
117 }

137 template <
138  class Poses,
139  class Weights,
140  class Pose = ranges::range_value_t<Poses>,
141  class Scalar = typename Pose::Scalar,
142  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
143 std::pair<Sophus::SE2<Scalar>, Eigen::Matrix3<Scalar>> estimate(Poses&& poses, Weights&& weights) {
144  auto translation_view = poses | ranges::views::transform([](const auto& pose) { return pose.translation(); });
145  auto weights_view = weights | ranges::views::common;
146  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
147  auto normalized_weights_view =
148  weights_view | ranges::views::transform([weights_sum](const auto& weight) { return weight / weights_sum; });
149 
150  // map sophus pose 2D pose into a 4D Eigen vector. Mapping exploits that Sophus stores the 2D transform
151  // as two elements for the linear translation, and two more for the orientation (in complex number form)
152  const auto pose_to_weighted_eigen_vector = [](const auto& pose, const auto& weight) {
153  return Eigen::Map<const Eigen::Vector4<Scalar>>{pose.data()} * weight;
154  };
155 
156  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
157  // that after averaging the complex representation of the orientation the resulting complex is not on the unit circle.
158  // This is expected and the value will be renormalized after having used the non-normal result to estimate the
159  // orientation autocovariance.
160  const Eigen::Vector4<Scalar> mean_pose_vector = std::transform_reduce(
161  poses.begin(), poses.end(), normalized_weights_view.begin(), Eigen::Vector4<Scalar>::Zero().eval(), std::plus{},
162  pose_to_weighted_eigen_vector);
163 
164  // Calculate the weighted pose estimation
165  Sophus::SE2<Scalar> estimated_pose = Eigen::Map<const Sophus::SE2<Scalar>>{mean_pose_vector.data()};
166 
167  Eigen::Matrix3<Scalar> covariance_matrix = Eigen::Matrix3<Scalar>::Zero();
168 
169  // Compute the covariance of the translation part.
170  covariance_matrix.template topLeftCorner<2, 2>() =
171  calculate_covariance(translation_view, normalized_weights_view, estimated_pose.translation());
172 
173  // Compute the orientation variance and re-normalize the rotation component.
174  if (estimated_pose.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
175  // Handle the case where both averages are too close to zero.
176  // Return zero yaw and infinite variance.
177  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
178  estimated_pose.so2() = Sophus::SO2<Scalar>{0.0};
179  } else {
180  // See circular standard deviation in
181  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
182  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(estimated_pose.so2().unit_complex().norm());
183  estimated_pose.so2().normalize();
184  }
185  return std::pair{estimated_pose, covariance_matrix};
186 }
