49 template <class Range, class WeightsRange, class Scalar>
50 Sophus::Matrix2<Scalar>
51 calculate_covariance(Range&& range, WeightsRange&& normalized_weights, const Sophus::Vector2<Scalar>& mean) {
52  auto translations_view = range | ranges::views::common;
53  auto normalized_weights_view = normalized_weights | ranges::views::common;
54 
55  auto calculate_weighted_sample_covariance = [mean](const auto& value, const auto& weight) {
56  const auto centered = value - mean;
57  return Sophus::Vector3<Scalar>{
58  weight * centered.x() * centered.x(), // weighted sample x autocovariance
59  weight * centered.x() * centered.y(), // weighted sample xy cross-covariance
60  weight * centered.y() * centered.y(), // weighted sample y autocovariance
61  };
62  };
63 
64  // calculate the averaging factor for the weighted covariance estimate
65  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
66  const auto squared_weight_sum = std::transform_reduce(
67  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
68  [](const auto& weight) { return (weight * weight); });
69 
70  // calculate the x autocovariance, xy cross-covariance, and y autocovariance
71  Sophus::Vector3<Scalar> coefficients =
72  std::transform_reduce(
73  translations_view.begin(), translations_view.end(), normalized_weights_view.begin(),
74  Sophus::Vector3<Scalar>::Zero().eval(), std::plus{}, calculate_weighted_sample_covariance) /
75  (1.0 - squared_weight_sum);
76 
77  // create the symmetric 2x2 translation covariance matrix from the coefficients
78  auto covariance_matrix = Sophus::Matrix2<Scalar>{};
79  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
80  return covariance_matrix;
81 }

121 template <
122  class Poses,
123  class Weights,
124  class Pose = ranges::range_value_t<Poses>,
125  class Scalar = typename Pose::Scalar,
126  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
127 std::pair<Sophus::SE2<Scalar>, Sophus::Matrix3<Scalar>> estimate(Poses&& poses, Weights&& weights) {
128  auto translation_view = poses | ranges::views::transform([](const auto& pose) { return pose.translation(); });
129  auto weights_view = weights | ranges::views::common;
130  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
131  auto normalized_weights_view =
132  weights_view | ranges::views::transform([weights_sum](const auto& weight) { return weight / weights_sum; });
133 
134  // map sophus pose 2D pose into a 4D Eigen vector. Mapping exploits that Sophus stores the 2D transform
135  // as two elements for the linear translation, and two more for the orientation (in complex number form)
136  const auto pose_to_weighted_eigen_vector = [](const auto& pose, const auto& weight) {
137  return Eigen::Map<const Sophus::Vector4<Scalar>>{pose.data()} * weight;
138  };
139 
140  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
141  // that after averaging the complex representation of the orientation the resulting complex is not on the unit circle.
142  // This is expected and the value will be renormalized after having used the non-normal result to estimate the
143  // orientation autocovariance.
144  const Sophus::Vector4<Scalar> mean_pose_vector = std::transform_reduce(
145  poses.begin(), poses.end(), normalized_weights_view.begin(), Sophus::Vector4<Scalar>::Zero().eval(), std::plus{},
146  pose_to_weighted_eigen_vector);
147 
148  // Calculate the weighted pose estimation
149  Sophus::SE2<Scalar> estimated_pose = Eigen::Map<const Sophus::SE2<Scalar>>{mean_pose_vector.data()};
150 
151  Sophus::Matrix3<Scalar> covariance_matrix = Sophus::Matrix3<Scalar>::Zero();
152 
153  // Compute the covariance of the translation part.
154  covariance_matrix.template topLeftCorner<2, 2>() =
155  calculate_covariance(translation_view, normalized_weights_view, estimated_pose.translation());
156 
157  // Compute the orientation variance and re-normalize the rotation component.
158  if (estimated_pose.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
159  // Handle the case where both averages are too close to zero.
160  // Return zero yaw and infinite variance.
161  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
162  estimated_pose.so2() = Sophus::SO2<Scalar>{0.0};
163  } else {
164  // See circular standard deviation in
165  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
166  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(estimated_pose.so2().unit_complex().norm());
167  estimated_pose.so2().normalize();
168  }
169  return std::pair{estimated_pose, covariance_matrix};
170 }
