97  [[nodiscard]] auto operator()(measurement_type&& points) const {
98  return [this, points = std::move(points)](const state_type& state) -> weight_type {
99  const auto beam = Ray2d{grid_, state, params_.beam_max_range};
100  const double n = 1. / (std::sqrt(2. * M_PI) * params_.sigma_hit);
101  return std::transform_reduce(
102  points.cbegin(), points.cend(), 0.0, std::plus{}, [this, &beam, n](const auto& point) {
103  // TODO(Ramiro): We're converting from range + bearing to cartesian points in the ROS node, but we want
104  // range
105  // + bearing here. We might want to make that conversion in the likelihood model instead, and let the
106  // measurement type be range, bearing instead of x, y.
107 
108  // Compute the range according to the measurement.
109  const double z = std::sqrt(point.first * point.first + point.second * point.second);
110 
111  // dirty hack to prevent SO2d from calculating the hypot again to normalize the vector.
112  auto beam_bearing = Sophus::SO2d{};
113  beam_bearing.data()[0] = point.first / z;
114  beam_bearing.data()[1] = point.second / z;
115 
116  // Compute range according to the map (raycasting).
117  const double z_mean = beam.cast(beam_bearing).value_or(params_.beam_max_range);
118  // 1: Correct range with local measurement noise.
119  const double eta_hit =
120  2. / (std::erf((params_.beam_max_range - z_mean) / (std::sqrt(2.) * params_.sigma_hit)) -
121  std::erf(-z_mean / (std::sqrt(2.) * params_.sigma_hit)));
122  const double d = (z - z_mean) / params_.sigma_hit;
123  double pz = params_.z_hit * eta_hit * n * std::exp(-(d * d) / 2.);
124 
125  // 2: Unexpected objects.
126  if (z < z_mean) {
127  const double eta_short = 1. / (1. - std::exp(-params_.lambda_short * z_mean));
128  pz += params_.z_short * params_.lambda_short * eta_short * std::exp(-params_.lambda_short * z);
129  }
130 
131  // 3 and 4: Max range return or random return.
132  if (z < params_.beam_max_range) {
133  pz += params_.z_rand / params_.beam_max_range;
134  } else {
135  pz += params_.z_max;
136  }
137 
138  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
139  // See https://github.com/Ekumen-OS/beluga/issues/153
140  return pz * pz * pz;
141  });
142  };
143  }
