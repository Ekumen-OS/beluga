123  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
124  const auto beam = Ray2d{grid_, state, params_.beam_max_range};
125  return std::transform_reduce(points_.cbegin(), points_.cend(), 0.0, std::plus{}, [this, &beam](const auto& point) {
126  // TODO(Ramiro): We're converting from range + bearing to cartesian points in the ROS node, but we want range +
127  // bearing here. We might want to make that conversion in the likelihood model instead, and let the measurement
128  // type be range, bearing instead of x, y.
129 
130  // Compute the range according to the measurement.
131  const double z = std::sqrt(point.first * point.first + point.second * point.second);
132 
133  // Compute range according to the map (raycasting).
134  const auto beam_bearing = Sophus::SO2d{point.first, point.second};
135  const double z_mean = beam.cast(beam_bearing).value_or(params_.beam_max_range);
136 
137  // 1: Correct range with local measurement noise.
138  const double eta_hit = 2. / (std::erf((params_.beam_max_range - z_mean) / (std::sqrt(2.) * params_.sigma_hit)) -
139  std::erf(-z_mean / (std::sqrt(2.) * params_.sigma_hit)));
140  const double d = (z - z_mean) / params_.sigma_hit;
141  const double n = 1. / (std::sqrt(2. * M_PI) * params_.sigma_hit);
142  double pz = params_.z_hit * eta_hit * n * std::exp(-(d * d) / 2.);
143 
144  // 2: Unexpected objects.
145  if (z < z_mean) {
146  const double eta_short = 1. / (1. - std::exp(-params_.lambda_short * z_mean));
147  pz += params_.z_short * params_.lambda_short * eta_short * std::exp(-params_.lambda_short * z);
148  }
149 
150  // 3 and 4: Max range return or random return.
151  if (z < params_.beam_max_range) {
152  pz += params_.z_rand / params_.beam_max_range;
153  } else {
154  pz += params_.z_max;
155  }
156 
157  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
158  // See https://github.com/Ekumen-OS/beluga/issues/153
159  return pz * pz * pz;
160  });
161  }
