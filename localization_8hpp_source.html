98 template <typename Mixin, typename Model>
99 class MotionMixin : public Mixin {
100  public:
101  using update_type = std::tuple_element_t<0, typename Model::control_type>;
102  using state_type = typename Model::state_type;
103 
104  template <typename... Args>
105  constexpr explicit MotionMixin(Model model, Args&&... args)
106  : Mixin(static_cast<decltype(args)>(args)...), model_(std::move(model)) {}
107 
108  template <class Generator>
109  [[nodiscard]] state_type apply_motion(const state_type& state, Generator& gen) {
110  if (state_sampling_function_.has_value()) {
111  return (*state_sampling_function_)(state, gen);
112  }
113  return state;
114  }
115 
116  void update_motion(const update_type& pose) { state_sampling_function_.emplace(model_(control_window_ << pose)); }
117 
118  private:
119  Model model_;
120  static constexpr auto kWindowSize = std::tuple_size_v<typename Model::control_type>;
121  RollingWindow<update_type, kWindowSize> control_window_;
122  using StateSamplingFunction =
123  decltype(std::declval<Model>()(std::declval<RollingWindow<update_type, kWindowSize>&>()));
124  std::optional<StateSamplingFunction> state_sampling_function_;
125 };
