156  template <class Range>
157  void initialize_particles(Range&& input) {
158  static_assert(std::is_convertible_v<ranges::range_value_t<Range>, particle_type>, "Invalid value type");
159  const std::size_t size = this->self().max_samples();
160  auto particles = input | ranges::views::take(size) | ranges::views::common;
161  particles_[1].reserve(size);
162  particles_[1].assign(ranges::begin(particles), ranges::end(particles));
163  std::swap(particles_[0], particles_[1]);
164  }
165 
167  void initialize_states(input_view_type input) final {
168  initialize_particles(input | ranges::views::transform(beluga::make_from_state<particle_type>));
169  }
