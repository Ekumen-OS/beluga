131  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
132  const auto transform = grid_.origin().inverse() * state;
133  const auto x_offset = transform.translation().x();
134  const auto y_offset = transform.translation().y();
135  const auto cos_theta = transform.so2().unit_complex().x();
136  const auto sin_theta = transform.so2().unit_complex().y();
137  const auto unknown_space_occupancy_prob = 1. / params_.max_laser_distance;
138  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
139  // See https://github.com/Ekumen-OS/beluga/issues/153
140  const auto unknown_space_occupancy_likelihood_cubed =
141  unknown_space_occupancy_prob * unknown_space_occupancy_prob * unknown_space_occupancy_prob;
142  return std::transform_reduce(
143  points_.cbegin(), points_.cend(), 1.0, std::plus{},
144  [this, x_offset, y_offset, cos_theta, sin_theta, unknown_space_occupancy_likelihood_cubed](const auto& point) {
145  // Transform the end point of the laser to the grid local coordinate system.
146  // Not using Eigen/Sophus because they make the routine x10 slower.
147  // See `benchmark_likelihood_field_model.cpp` for reference.
148  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
149  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
150  // for performance, we store the likelihood already elevated to the cube
151  return likelihood_field_.data_near(x, y).value_or(unknown_space_occupancy_likelihood_cubed);
152  });
153  }

177  static ValueGrid2<double> make_likelihood_field(const LikelihoodFieldModelParam& params, const OccupancyGrid& grid) {
178  const auto squared_distance = [&grid,
179  squared_max_distance = params.max_obstacle_distance * params.max_obstacle_distance](
180  std::size_t first, std::size_t second) {
181  return std::min((grid.coordinates_at(first) - grid.coordinates_at(second)).squaredNorm(), squared_max_distance);
182  };
183 
184  const auto neighborhood = [&grid](std::size_t index) { return grid.neighborhood4(index); };
185 
186  const auto distance_map = nearest_obstacle_distance_map(grid.obstacle_data(), squared_distance, neighborhood);
187 
188  const auto to_likelihood = [amplitude =
189  params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi())),
190  two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit,
191  offset = params.z_random / params.max_laser_distance](double squared_distance) {
192  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
193  };
194 
195  // we store the likelihood elevated to the cube to save a few runtime computations
196  // when calculating the importance weight
197  const auto to_the_cube = [](auto likelihood) { return likelihood * likelihood * likelihood; };
198 
199  auto likelihood_data = distance_map | ranges::views::transform(to_likelihood) |
200  ranges::views::transform(to_the_cube) | ranges::to<std::vector>;
201  return ValueGrid2<double>{std::move(likelihood_data), grid.width(), grid.resolution()};
202  }
