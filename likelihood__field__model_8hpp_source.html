127  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
128  const auto transform = grid_.origin().inverse() * state;
129  const auto x_offset = transform.translation().x();
130  const auto y_offset = transform.translation().y();
131  const auto cos_theta = transform.so2().unit_complex().x();
132  const auto sin_theta = transform.so2().unit_complex().y();
133  const auto unknown_space_occupancy_prob = 1. / params_.max_laser_distance;
134  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
135  // See https://github.com/Ekumen-OS/beluga/issues/153
136  const auto unknown_space_occupancy_likelihood_cubed =
137  unknown_space_occupancy_prob * unknown_space_occupancy_prob * unknown_space_occupancy_prob;
138  return std::transform_reduce(
139  points_.cbegin(), points_.cend(), 1.0, std::plus{},
140  [this, x_offset, y_offset, cos_theta, sin_theta, unknown_space_occupancy_likelihood_cubed](const auto& point) {
141  // Transform the end point of the laser to the grid local coordinate system.
142  // Not using Eigen/Sophus because they make the routine x10 slower.
143  // See `benchmark_likelihood_field_model.cpp` for reference.
144  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
145  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
146  // for performance, we store the likelihood already elevated to the cube
147  return likelihood_field_.data_near(x, y).value_or(unknown_space_occupancy_likelihood_cubed);
148  });
149  }

173  static ValueGrid2<double> make_likelihood_field(const LikelihoodFieldModelParam& params, const OccupancyGrid& grid) {
174  const auto squared_distance = [&grid,
175  squared_max_distance = params.max_obstacle_distance * params.max_obstacle_distance](
176  std::size_t first, std::size_t second) {
177  return std::min((grid.coordinates_at(first) - grid.coordinates_at(second)).squaredNorm(), squared_max_distance);
178  };
179 
180  const auto neighborhood = [&grid](std::size_t index) { return grid.neighborhood4(index); };
181 
182  const auto distance_map = nearest_obstacle_distance_map(grid.obstacle_data(), squared_distance, neighborhood);
183 
184  const auto to_likelihood = [amplitude =
185  params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi())),
186  two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit,
187  offset = params.z_random / params.max_laser_distance](double squared_distance) {
188  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
189  };
190 
191  // we store the likelihood elevated to the cube to save a few runtime computations
192  // when calculating the importance weight
193  const auto to_the_cube = [](auto likelihood) { return likelihood * likelihood * likelihood; };
194 
195  auto likelihood_data = distance_map | ranges::views::transform(to_likelihood) |
196  ranges::views::transform(to_the_cube) | ranges::to<std::vector>;
197  return ValueGrid2<double>{std::move(likelihood_data), grid.width(), grid.resolution()};
198  }
