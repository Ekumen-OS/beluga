108  [[nodiscard]] auto operator()(measurement_type&& points) const {
109  return [this, points = std::move(points)](const state_type& state) -> weight_type {
110  const auto transform = world_to_likelihood_field_transform_ * state;
111  const auto x_offset = transform.translation().x();
112  const auto y_offset = transform.translation().y();
113  const auto cos_theta = transform.so2().unit_complex().x();
114  const auto sin_theta = transform.so2().unit_complex().y();
115  const auto unknown_space_occupancy_prob = 1. / params_.max_laser_distance;
116  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
117  // See https://github.com/Ekumen-OS/beluga/issues/153
118  const auto unknown_space_occupancy_likelihood_cubed =
119  unknown_space_occupancy_prob * unknown_space_occupancy_prob * unknown_space_occupancy_prob;
120  return std::transform_reduce(
121  points.cbegin(), points.cend(), 1.0, std::plus{},
122  [this, x_offset, y_offset, cos_theta, sin_theta,
123  unknown_space_occupancy_likelihood_cubed](const auto& point) {
124  // Transform the end point of the laser to the grid local coordinate system.
125  // Not using Eigen/Sophus because they make the routine x10 slower.
126  // See `benchmark_likelihood_field_model.cpp` for reference.
127  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
128  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
129  // for performance, we store the likelihood already elevated to the cube
130  return likelihood_field_.data_near(x, y).value_or(unknown_space_occupancy_likelihood_cubed);
131  });
132  };
133  }

151  static ValueGrid2<double> make_likelihood_field(const LikelihoodFieldModelParam& params, const OccupancyGrid& grid) {
152  const auto squared_distance = [&grid,
153  squared_max_distance = params.max_obstacle_distance * params.max_obstacle_distance](
154  std::size_t first, std::size_t second) {
155  return std::min((grid.coordinates_at(first) - grid.coordinates_at(second)).squaredNorm(), squared_max_distance);
156  };
157 
158  const auto neighborhood = [&grid](std::size_t index) { return grid.neighborhood4(index); };
159 
160  const auto distance_map = nearest_obstacle_distance_map(grid.obstacle_data(), squared_distance, neighborhood);
161 
162  const auto to_likelihood = [amplitude =
163  params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi())),
164  two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit,
165  offset = params.z_random / params.max_laser_distance](double squared_distance) {
166  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
167  };
168 
169  // we store the likelihood elevated to the cube to save a few runtime computations
170  // when calculating the importance weight
171  const auto to_the_cube = [](auto likelihood) { return likelihood * likelihood * likelihood; };
172 
173  auto likelihood_data = distance_map | ranges::views::transform(to_likelihood) |
174  ranges::views::transform(to_the_cube) | ranges::to<std::vector>;
175  return ValueGrid2<double>{std::move(likelihood_data), grid.width(), grid.resolution()};
176  }
