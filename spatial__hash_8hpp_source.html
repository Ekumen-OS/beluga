44 template <std::size_t N, std::size_t I>
45 constexpr std::size_t floor_and_shift(double value) {
46  // Compute the largest integer value not greater than value.
47  auto signed_value = static_cast<std::intmax_t>(std::floor(value));
48  // Compute the smallest unsigned integer equal to the source value modulo 2^n
49  // (where n is the number of bits used to represent the destination type).
50  auto unsigned_value = static_cast<std::uintmax_t>(signed_value);
51  // Create a fixed-size sequence of N bits and perform a binary shift left I * N positions.
52  return std::bitset<N>{unsigned_value}.to_ullong() << (I * N);
53 }

65 template <class T, std::size_t... Ids>
66 constexpr std::size_t hash_impl(
67  const T& value,
68  const std::array<double, sizeof...(Ids)>& resolution,
69  [[maybe_unused]] std::index_sequence<Ids...> index_sequence) {
70  constexpr auto kBits = std::numeric_limits<std::size_t>::digits / sizeof...(Ids);
71  return (detail::floor_and_shift<kBits, Ids>(std::get<Ids>(value) / resolution[Ids]) | ...);
