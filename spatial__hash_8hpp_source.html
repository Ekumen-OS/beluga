44 template <std::size_t N, std::size_t I>
45 constexpr std::size_t floor_and_fibo_hash(double value) {
46  static_assert(std::is_same_v<std::size_t, std::uint64_t>);
47  constexpr auto kFib64 = 11400714819323198485LLU; // golden ratio for 64 bits
48  // floor the value and convert to integer
49  const auto signed_value = static_cast<std::int64_t>(std::floor(value));
50  // work with unsigned from now on
51  const auto unsigned_value = static_cast<std::uint64_t>(signed_value);
52  // spread number information all through the 64 bits using the fibonacci hash
53  const auto div_hashed_value = kFib64 * unsigned_value;
54  // rotate bits to avoid aliasing between different values of I
55  if constexpr (N * I != 0) {
56  const auto left_hash = (div_hashed_value << N * I);
57  const auto right_hash = (div_hashed_value >> (64 - N * I));
58  return left_hash | right_hash;
59  } else {
60  return div_hashed_value;
61  }
62 }

74 template <class T, std::size_t... Ids>
75 constexpr std::size_t hash_impl(
76  const T& value,
77  const std::array<double, sizeof...(Ids)>& resolution,
78  [[maybe_unused]] std::index_sequence<Ids...> index_sequence) {
79  constexpr auto kBits = std::numeric_limits<std::size_t>::digits / sizeof...(Ids);
