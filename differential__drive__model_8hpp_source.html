103  template <class Generator>
104  [[nodiscard]] state_type apply_motion(const state_type& state, Generator& gen) const {
105  static thread_local auto distribution = std::normal_distribution<double>{};
106  const auto first_rotation = Sophus::SO2d{distribution(gen, first_rotation_params_)};
107  const auto translation = Eigen::Vector2d{distribution(gen, translation_params_), 0.0};
108  const auto second_rotation = Sophus::SO2d{distribution(gen, second_rotation_params_)};
109  return state * Sophus::SE2d{first_rotation, Eigen::Vector2d{0.0, 0.0}} * Sophus::SE2d{second_rotation, translation};
110  }

113  void update_motion(const update_type& pose) final {
114  if (last_pose_) {
115  const auto translation = pose.translation() - last_pose_.value().translation();
116  const double distance = translation.norm();
117  const double distance_variance = distance * distance;
118 
119  const auto& previous_orientation = last_pose_.value().so2();
120  const auto& current_orientation = pose.so2();
121  const auto first_rotation =
122  distance > params_.distance_threshold
123  ? Sophus::SO2d{std::atan2(translation.y(), translation.x())} * previous_orientation.inverse()
124  : Sophus::SO2d{};
125  const auto second_rotation = current_orientation * previous_orientation.inverse() * first_rotation.inverse();
126 
127  {
128  first_rotation_params_ = DistributionParam{
129  first_rotation.log(), std::sqrt(
130  params_.rotation_noise_from_rotation * rotation_variance(first_rotation) +
131  params_.rotation_noise_from_translation * distance_variance)};
132  translation_params_ = DistributionParam{
133  distance, std::sqrt(
134  params_.translation_noise_from_translation * distance_variance +
135  params_.translation_noise_from_rotation *
136  (rotation_variance(first_rotation) + rotation_variance(second_rotation)))};
137  second_rotation_params_ = DistributionParam{
138  second_rotation.log(), std::sqrt(
139  params_.rotation_noise_from_rotation * rotation_variance(second_rotation) +
140  params_.rotation_noise_from_translation * distance_variance)};
141  }
142  }
143  last_pose_ = pose;
144  }
145 
146  private:
147  using DistributionParam = typename std::normal_distribution<double>::param_type;

152  DistributionParam first_rotation_params_{0.0, 0.0};
153  DistributionParam second_rotation_params_{0.0, 0.0};
154  DistributionParam translation_params_{0.0, 0.0};

156  static double rotation_variance(const Sophus::SO2d& rotation) {
157  // Treat backward and forward motion symmetrically for the noise models.
158  static const auto kFlippingRotation = Sophus::SO2d{Sophus::Constants<double>::pi()};
159  const auto flipped_rotation = rotation * kFlippingRotation;
160  const auto delta = std::min(std::abs(rotation.log()), std::abs(flipped_rotation.log()));
161  return delta * delta;
162  }
