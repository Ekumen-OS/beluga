107  template <class Generator>
108  [[nodiscard]] state_type apply_motion(const state_type& state, Generator& gen) const {
109  static thread_local auto distribution = std::normal_distribution<double>{};
110  const auto lock = std::shared_lock<std::shared_mutex>{params_mutex_};
111  // This is an implementation based on the same set of parameters that is used in
112  // nav2's omni_motion_model. To simplify the implementation, the following
113  // variable substitutions were performed:
114  // - first_rotation = delta_bearing - previous_orientation
115  // - second_rotation = delta_rot_hat - first_rotation
116  const auto second_rotation = Sophus::SO2d{distribution(gen, rotation_params_)} * first_rotation_.inverse();
117  const auto translation = Eigen::Vector2d{
118  distribution(gen, translation_params_),
119  -1.0 * distribution(gen, strafe_params_),
120  };
121  return state * Sophus::SE2d{first_rotation_, Eigen::Vector2d{0.0, 0.0}} *
122  Sophus::SE2d{second_rotation, translation};
123  }

126  void update_motion(const update_type& pose) final {
127  if (last_pose_) {
128  const auto translation = pose.translation() - last_pose_.value().translation();
129  const double distance = translation.norm();
130  const double distance_variance = distance * distance;
131 
132  const auto& previous_orientation = last_pose_.value().so2();
133  const auto& current_orientation = pose.so2();
134  const auto rotation = current_orientation * previous_orientation.inverse();
135 
136  {
137  const auto lock = std::lock_guard<std::shared_mutex>{params_mutex_};
138  first_rotation_ =
139  distance > params_.distance_threshold
140  ? Sophus::SO2d{std::atan2(translation.y(), translation.x())} * previous_orientation.inverse()
141  : Sophus::SO2d{0.0};
142 
143  rotation_params_ = DistributionParam{
144  rotation.log(), std::sqrt(
145  params_.rotation_noise_from_rotation * rotation_variance(rotation) +
146  params_.rotation_noise_from_translation * distance_variance)};
147  translation_params_ = DistributionParam{
148  distance, std::sqrt(
149  params_.translation_noise_from_translation * distance_variance +
150  params_.translation_noise_from_rotation * rotation_variance(rotation))};
151  strafe_params_ = DistributionParam{
152  0.0, std::sqrt(
153  params_.strafe_noise_from_translation * distance_variance +
154  params_.translation_noise_from_rotation * rotation_variance(rotation))};
155  }
156  }
157  last_pose_ = pose;
158  }

178  static double rotation_variance(const Sophus::SO2d& rotation) {
179  // Treat backward and forward motion symmetrically for the noise models.
180  const auto flipped_rotation = rotation * Sophus::SO2d{Sophus::Constants<double>::pi()};
181  const auto delta = std::min(std::abs(rotation.log()), std::abs(flipped_rotation.log()));
182  return delta * delta;
