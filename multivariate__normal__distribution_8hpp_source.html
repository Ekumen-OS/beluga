102  [[nodiscard]] static Matrix make_transform(const Matrix& covariance) {
103  // For more information about the method used to generate correlated normal vectors
104  // from independent normal distributions, see:
105  // https://en.wikipedia.org/wiki/Multivariate_normal_distribution#Drawing_values_from_the_distribution
106  // Gentle, J. E. (2009). Computational Statistics. Statistics and Computing. New York: Springer. pp. 315â€“316.
107  // https://doi.org/10.1007%2F978-0-387-98144-4
108  if (!covariance.isApprox(covariance.transpose())) {
109  throw std::runtime_error("Invalid covariance matrix, it is not symmetric.");
110  }
111  const auto solver = Eigen::SelfAdjointEigenSolver<Matrix>{covariance};
112  if (solver.info() != Eigen::Success) {
113  throw std::runtime_error("Invalid covariance matrix, eigen solver failed.");
114  }
115  const auto eigenvalues = solver.eigenvalues();
116  if ((eigenvalues.array() < 0.0).any()) {
117  throw std::runtime_error("Invalid covariance matrix, it has negative eigenvalues.");
118  }
119  return solver.eigenvectors() * eigenvalues.cwiseSqrt().asDiagonal();
120  }
121  };
