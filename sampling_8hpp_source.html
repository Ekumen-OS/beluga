245  template <class Generator>
246  [[nodiscard]] auto generate_samples_from_particles(Generator& engine) {
247  auto&& particles = this->self().particles();
248  static_assert(is_particle_range_v<decltype(particles)>);
249 
250  const double total_weight = ranges::accumulate(beluga::views::weights(particles), 0.0);
251  const double average_weight = total_weight / static_cast<double>(particles.size());
252  const double random_state_probability =
253  std::clamp(1.0 - fast_filter_(average_weight) / slow_filter_(average_weight), 0.0, 1.0);
254 
255  if (random_state_probability > 0.0) {
256  fast_filter_.reset();
257  slow_filter_.reset();
258  }
259 
260  auto make_random_particle = [this, &engine] {
261  using particle_type = typename Mixin::self_type::particle_type;
262  return beluga::make_from_state<particle_type>(this->self().make_random_state(engine));
263  };
264  return particles | //
265  beluga::views::sample(engine) | //
266  beluga::views::random_intersperse(std::move(make_random_particle), random_state_probability, engine);
267  }
268 
269  private:
270  ExponentialFilter slow_filter_;
271  ExponentialFilter fast_filter_;
272 };
