153 template <class Range, class Weights, class RandomNumberGenerator>
154 auto random_sample(const Range& samples, const Weights& weights, RandomNumberGenerator& generator) {
155  auto weights_begin = std::begin(weights);
156  auto weights_end = std::end(weights);
157  using difference_type = decltype(weights_end - weights_begin);
158  return [&generator, first = std::begin(samples),
159  distribution = std::discrete_distribution<difference_type>{weights_begin, weights_end}]() mutable {
160  return *(first + distribution(generator));
161  };
162 }

215 inline auto kld_condition(std::size_t min, double epsilon, double z = 3.) {
216  auto target_size = [two_epsilon = 2 * epsilon, z](std::size_t k) {
217  if (k <= 2U) {
218  return std::numeric_limits<std::size_t>::max();
219  }
220  double common = 2. / static_cast<double>(9 * (k - 1));
221  double base = 1. - common + std::sqrt(common) * z;
222  double result = (static_cast<double>(k - 1) / two_epsilon) * base * base * base;
223  return static_cast<std::size_t>(std::ceil(result));
224  };
225 
226  return [=, count = 0ULL, buckets = std::unordered_set<std::size_t>{}](std::size_t hash) mutable {
227  count++;
228  buckets.insert(hash);
229  return count <= min || count <= target_size(buckets.size());
230  };
231 }

364  template <class Generator>
365  [[nodiscard]] auto generate_samples_from_particles(Generator& gen) {
366  auto weights = this->self().weights() | ranges::views::common;
367  double total_weight = std::reduce(std::begin(weights), std::end(weights), 0.);
368  double average_weight = total_weight / static_cast<double>(weights.size());
369  double random_state_probability = std::max(0., 1. - fast_filter_(average_weight) / slow_filter_(average_weight));
370 
371  if (random_state_probability > 0.) {
372  fast_filter_.reset();
373  slow_filter_.reset();
374  }
375 
376  return ranges::views::generate(random_select(
377  [this, &gen]() { return this->self().make_random_state(gen); },
378  random_sample(this->self().states(), weights, gen), gen, random_state_probability));
379  }
380 
381  private:
382  ExponentialFilter slow_filter_;
383  ExponentialFilter fast_filter_;
384 };
