264  template <class Generator>
265  [[nodiscard]] auto generate_samples_from_particles(Generator& engine) {
266  auto&& particles = this->self().particles();
267  static_assert(is_particle_range_v<decltype(particles)>);
268 
269  const double total_weight = ranges::accumulate(beluga::views::weights(particles), 0.0);
270  const double average_weight = total_weight / static_cast<double>(particles.size());
271  const double random_state_probability =
272  std::clamp(1.0 - fast_filter_(average_weight) / slow_filter_(average_weight), 0.0, 1.0);
273 
274  if (random_state_probability > 0.0) {
275  fast_filter_.reset();
276  slow_filter_.reset();
277  }
278 
279  auto make_random_particle = [this, &engine] {
280  using particle_type = typename Mixin::self_type::particle_type;
281  return beluga::make_from_state<particle_type>(this->self().make_random_state(engine));
282  };
283  return particles | //
284  beluga::views::sample(engine) | //
285  beluga::views::random_intersperse(std::move(make_random_particle), random_state_probability, engine);
286  }
287 
288  private:
289  ExponentialFilter slow_filter_;
290  ExponentialFilter fast_filter_;
291 };
